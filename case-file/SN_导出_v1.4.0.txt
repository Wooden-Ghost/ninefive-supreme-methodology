    function setupPanelEvents(panel, titleBar) {
        let isDragging = false;
        let startX = 0;
        let startY = 0;
        let startLeft = 0;
        let startTop = 0;

        titleBar.addEventListener('mousedown', function (e) {
            if (e.target.closest('#close-btn')) return;

            isDragging = true;
            startX = e.clientX;
            startY = e.clientY;
            startLeft = panel.offsetLeft;
            startTop = panel.offsetTop;

            document.addEventListener('mousemove', onDrag);
            document.addEventListener('mouseup', stopDrag);
        });

        function onDrag(e) {
            if (!isDragging) return;

            const dx = e.clientX - startX;
            const dy = e.clientY - startY;

            let newLeft = startLeft + dx;
            let newTop = startTop + dy;

            const maxX = window.innerWidth - panel.offsetWidth;
            const maxY = window.innerHeight - panel.offsetHeight;

            newLeft = Math.max(0, Math.min(newLeft, maxX));
            newTop = Math.max(0, Math.min(newTop, maxY));

            panel.style.left = `${newLeft}px`;
            panel.style.top = `${newTop}px`;
            panel.style.right = 'auto';
            panel.style.bottom = 'auto';
        }

        function stopDrag() {
            isDragging = false;
            document.removeEventListener('mousemove', onDrag);
            document.removeEventListener('mouseup', stopDrag);
        }

        document.getElementById('close-btn').addEventListener('click', function () {
            panel.style.display = 'none';
        });
    }

    function waitForCondition(fn, timeout = 10000, interval = 250) {
        return new Promise((resolve, reject) => {
            const start = Date.now();

            const check = () => {
                if (!window.isExportRunning && window.isExportRunning !== undefined) {
                    reject(new Error('任务已停止'));
                    return;
                }

                try {
                    const result = fn();
                    if (result) {
                        resolve(result);
                        return;
                    }
                } catch (e) {}

                if (Date.now() - start > timeout) {
                    reject(new Error('等待超时'));
                    return;
                }

                setTimeout(check, interval);
            };

            check();
        });
    }

    function isVisible(el) {
        if (!el) return false;
        const style = window.getComputedStyle(el);
        return style.display !== 'none' &&
            style.visibility !== 'hidden' &&
            el.offsetParent !== null;
    }

    function isEnabled(el) {
        if (!el) return false;
        return !el.disabled && !el.hasAttribute('disabled') && el.getAttribute('aria-disabled') !== 'true';
    }

    function safeClick(element) {
        if (!element) return false;

        try {
            element.scrollIntoView({ block: 'center', inline: 'center', behavior: 'auto' });
        } catch (e) {}

        try {
            element.dispatchEvent(new MouseEvent('mouseover', { bubbles: true, cancelable: true }));
            element.dispatchEvent(new MouseEvent('mousedown', { bubbles: true, cancelable: true }));
            element.dispatchEvent(new MouseEvent('mouseup', { bubbles: true, cancelable: true }));
            element.dispatchEvent(new MouseEvent('click', { bubbles: true, cancelable: true }));
            return true;
        } catch (e) {
            try {
                element.click();
                return true;
            } catch (e2) {
                return false;
            }
        }
    }

    function limitText(text, maxLen = 260) {
        const s = String(text || '').replace(/\s+/g, ' ').trim();
        if (s.length <= maxLen) return s;
        return s.slice(0, maxLen) + '...';
    }

    function getElementText(el) {
        return (el?.textContent || '').replace(/\s+/g, ' ').trim();
    }

    function buildStableSelector(el) {
        if (!el) return null;

        if (el.id) {
            return `#${CSS.escape(el.id)}`;
        }

        const dataTest = el.getAttribute('data-test');
        if (dataTest) {
            return `[data-test="${CSS.escape(dataTest)}"]`;
        }

        const trackingId = el.getAttribute('plerdy-tracking-id');
        if (trackingId) {
            return `[plerdy-tracking-id="${trackingId}"]`;
        }

        const classes = [...el.classList].filter(Boolean).slice(0, 4);
        if (classes.length) {
            return `${el.tagName.toLowerCase()}.${classes.map(c => CSS.escape(c)).join('.')}`;
        }

        return el.tagName.toLowerCase();
    }

    function getCssPath(el) {
        if (!el || el.nodeType !== 1) return '';
        const parts = [];
        let node = el;
        while (node && node.nodeType === 1 && node !== document.body) {
            let part = node.tagName.toLowerCase();
            if (node.id) {
                part += `#${node.id}`;
                parts.unshift(part);
                break;
            }
            const cls = [...node.classList].filter(Boolean).slice(0, 2);
            if (cls.length) {
                part += '.' + cls.join('.');
            }
            const parent = node.parentElement;
            if (parent) {
                const sameTagSiblings = [...parent.children].filter(child => child.tagName === node.tagName);
                if (sameTagSiblings.length > 1) {
                    part += `:nth-of-type(${sameTagSiblings.indexOf(node) + 1})`;
                }
            }
            parts.unshift(part);
            node = node.parentElement;
        }
        return parts.join(' > ');
    }

    function getEmailSearchingBlock() {
        const blocks = document.querySelectorAll('.check-verification-block');

        for (const block of blocks) {
            if (!isVisible(block)) continue;

            const text = (block.textContent || '').trim();
            const hasAnimatedLoadingIcon = !!block.querySelector('use[xlink\\:href="#loading_icon__animated"]');
            const hasSearchingText = text.includes('正在搜索邮箱地址');

            if (hasAnimatedLoadingIcon && hasSearchingText) {
                return block;
            }
        }

        return null;
    }

    function collectActiveEmailSearchingElements() {
        const matches = [];
        const seen = new Set();

        function pushMatch(el, source) {
            if (!el || seen.has(el) || !isVisible(el)) return;
            const text = getElementText(el);
            if (!text.includes('正在搜索邮箱地址')) return;
            seen.add(el);
            matches.push({
                element: el,
                source,
                text,
                selector: buildStableSelector(el),
                cssPath: getCssPath(el)
            });
        }

        document.querySelectorAll('.check-verification-block').forEach(block => {
            const text = getElementText(block);
            const hasAnimatedLoadingIcon = !!block.querySelector('use[xlink\\:href="#loading_icon__animated"]');
            if (hasAnimatedLoadingIcon && text.includes('正在搜索邮箱地址')) {
                pushMatch(block, 'top-block');
            }
        });

        document.querySelectorAll('.generic__searching, .stop-drag, .row-info, td.row__cell--relative').forEach(el => {
            const text = getElementText(el);
            if (!text.includes('正在搜索邮箱地址')) return;

            const searchingNode = el.classList.contains('generic__searching')
                ? el
                : (el.querySelector('.generic__searching') || el);

            pushMatch(searchingNode, 'row-inline');
        });

        document.querySelectorAll('div, span').forEach(el => {
            if (seen.has(el) || !isVisible(el)) return;
            const text = getElementText(el);
            if (!text.includes('正在搜索邮箱地址')) return;
            if (text.length > 80) return;
            pushMatch(el, 'text-fallback');
        });

        return matches;
    }

    function collectSearchStatusCandidates() {
        const candidateSet = new Set();
        const keywordSelectors = [
            '.check-verification-block',
            '[data-test*="search"]',
            '[class*="search"]',
            '[class*="verification"]',
            '[class*="loading"]',
            '[class*="alert"]'
        ];

        keywordSelectors.forEach(selector => {
            document.querySelectorAll(selector).forEach(el => {
                if (isVisible(el)) candidateSet.add(el);
            });
        });

        document.querySelectorAll('div, section, article, span, p').forEach(el => {
            if (!isVisible(el)) return;
            const text = getElementText(el);
            if (!text) return;
            if (text.length > 180) return;
            if (
                text.includes('正在搜索邮箱地址') ||
                text.includes('搜索邮箱地址') ||
                text.includes('搜索邮箱') ||
                text.includes('邮箱地址') ||
                text.includes('搜索中')
            ) {
                candidateSet.add(el);
            }
        });

        const candidates = [];
        candidateSet.forEach(el => {
            const text = getElementText(el);
            const hasAnimatedLoadingIcon = !!el.querySelector('use[xlink\\:href="#loading_icon__animated"]');
            const hasStaticLoadingIcon = !!el.querySelector('use[xlink\\:href*="#loading_icon"]');
            const className = (el.className || '').toString();
            const dataTest = el.getAttribute('data-test') || '';

            let score = 0;
            if (text.includes('正在搜索邮箱地址')) score += 100;
            if (text.includes('搜索邮箱地址')) score += 60;
            if (text.includes('搜索邮箱')) score += 40;
            if (text.includes('邮箱地址')) score += 20;
            if (text.includes('搜索中')) score += 15;
            if (hasAnimatedLoadingIcon) score += 50;
            if (hasStaticLoadingIcon) score += 10;
            if (/check-verification-block/i.test(className)) score += 20;
            if (/search|verification|loading/i.test(className)) score += 10;
            if (/search|verification/i.test(dataTest)) score += 10;

            const rect = el.getBoundingClientRect();
            candidates.push({
                element: el,
                score,
                text: limitText(text, 320),
                dataTest,
                className: limitText(className, 200),
                selector: buildStableSelector(el),
                cssPath: getCssPath(el),
                hasAnimatedLoadingIcon,
                hasStaticLoadingIcon,
                rect: {
                    x: Math.round(rect.x),
                    y: Math.round(rect.y),
                    w: Math.round(rect.width),
                    h: Math.round(rect.height)
                },
                html: limitText(el.outerHTML || '', 800)
            });
        });

        candidates.sort((a, b) => b.score - a.score);
        return candidates.slice(0, 20);
    }

    function scanSearchStatusNow() {
        const exactBlock = getEmailSearchingBlock();
        const candidates = collectSearchStatusCandidates();
        const best = exactBlock
            ? {
                element: exactBlock,
                score: 999,
                text: limitText(getElementText(exactBlock), 320),
                dataTest: exactBlock.getAttribute('data-test') || '',
                className: limitText((exactBlock.className || '').toString(), 200),
                selector: buildStableSelector(exactBlock),
                cssPath: getCssPath(exactBlock),
                hasAnimatedLoadingIcon: !!exactBlock.querySelector('use[xlink\\:href="#loading_icon__animated"]'),
                hasStaticLoadingIcon: !!exactBlock.querySelector('use[xlink\\:href*="#loading_icon"]'),
                rect: (() => {
                    const rect = exactBlock.getBoundingClientRect();
                    return {
                        x: Math.round(rect.x),
                        y: Math.round(rect.y),
                        w: Math.round(rect.width),
                        h: Math.round(rect.height)
                    };
                })(),
                html: limitText(exactBlock.outerHTML || '', 800)
            }
            : (candidates[0] || null);

        const found = !!exactBlock || (!!best && best.score >= 70);

        return {
            found,
            best,
            candidates,
            exactMatch: !!exactBlock,
            scannedAt: new Date().toLocaleString(),
            url: location.href,
            title: document.title
        };
    }

    async function runSearchStatusRecon() {
        if (reconState.running) {
            logMessage('搜索状态侦察已在运行中', 'warn');
            return;
        }

        resetReconState();
        reconState.running = true;
        reconState.startedAt = new Date().toLocaleString();
        reconState.result = 'running';
        appendReconLine('Snov Search Status Recon Report');
        appendReconLine(`generatedAt: ${reconState.startedAt}`);
        appendReconLine(`timeoutMs: ${SEARCH_RECON_TIMEOUT_MS}`);
        appendReconLine(`startUrl: ${location.href}`);
        appendReconLine(`title: ${document.title}`);
        appendReconLine('');

        logMessage('开始侦察搜索状态，最长等待 5 分钟...');

        const start = Date.now();
        let lastHeartbeat = 0;

        try {
            while (Date.now() - start < SEARCH_RECON_TIMEOUT_MS) {
                const snapshot = scanSearchStatusNow();
                reconState.latestSnapshot = snapshot;
                reconState.latestCandidates = snapshot.candidates;
                reconState.bestCandidate = snapshot.best;

                if (snapshot.found) {
                    reconState.result = 'found';
                    reconState.endedAt = new Date().toLocaleString();
                    reconState.lastMessage = snapshot.exactMatch ? '命中精确搜索状态块' : '命中高分候选搜索状态块';

                    appendReconLine(`[FOUND] ${reconState.endedAt}`);
                    appendReconLine(`message: ${reconState.lastMessage}`);
                    appendReconLine(`url: ${snapshot.url}`);
                    appendReconLine(`title: ${snapshot.title}`);
                    appendReconLine('');
                    appendReconLine(formatCandidateBlock(snapshot.best, 'bestCandidate'));
                    appendReconLine('');
                    appendReconLine(formatCandidateList(snapshot.candidates));

                    logMessage(`已侦察到搜索状态：${reconState.lastMessage}`, 'success');
                    return true;
                }

                const now = Date.now();
                if (now - lastHeartbeat >= 10000) {
                    const elapsedSec = Math.floor((now - start) / 1000);
                    const topScore = snapshot.best ? snapshot.best.score : 0;
                    logMessage(`侦察中... 已等待 ${elapsedSec} 秒，当前最高候选分数 ${topScore}`);
                    lastHeartbeat = now;
                }

                await wait(SEARCH_RECON_INTERVAL_MS);
            }

            reconState.result = 'timeout';
            reconState.endedAt = new Date().toLocaleString();
            reconState.lastMessage = '5 分钟内未命中可确认的搜索状态块';

            appendReconLine(`[TIMEOUT] ${reconState.endedAt}`);
            appendReconLine(`message: ${reconState.lastMessage}`);
            appendReconLine(`url: ${location.href}`);
            appendReconLine(`title: ${document.title}`);
            appendReconLine('');

            if (reconState.bestCandidate) {
                appendReconLine(formatCandidateBlock(reconState.bestCandidate, 'bestCandidateWhenTimeout'));
                appendReconLine('');
            }
            appendReconLine(formatCandidateList(reconState.latestCandidates));

            logMessage('搜索状态侦察超时：5 分钟内未命中明确状态', 'error');
            return false;
        } catch (error) {
            reconState.result = 'error';
            reconState.endedAt = new Date().toLocaleString();
            reconState.lastMessage = error.message || String(error);
            appendReconLine(`[ERROR] ${reconState.endedAt}`);
            appendReconLine(`message: ${reconState.lastMessage}`);
            logMessage(`搜索状态侦察失败：${reconState.lastMessage}`, 'error');
            return false;
        } finally {
            reconState.running = false;
        }
    }

    function formatCandidateBlock(candidate, title = 'candidate') {
        if (!candidate) return `${title}: (none)`;
        return [
            `===== ${title} =====`,
            `score: ${candidate.score}`,
            `text: ${candidate.text || ''}`,
            `selector: ${candidate.selector || ''}`,
            `data-test: ${candidate.dataTest || ''}`,
            `class: ${candidate.className || ''}`,
            `cssPath: ${candidate.cssPath || ''}`,
            `loadingAnimated: ${candidate.hasAnimatedLoadingIcon ? 'yes' : 'no'}`,
            `loadingIcon: ${candidate.hasStaticLoadingIcon ? 'yes' : 'no'}`,
            `rect: x=${candidate.rect?.x ?? ''}, y=${candidate.rect?.y ?? ''}, w=${candidate.rect?.w ?? ''}, h=${candidate.rect?.h ?? ''}`,
            `html: ${candidate.html || ''}`
        ].join('\n');
    }

    function formatCandidateList(candidates) {
        const list = Array.isArray(candidates) ? candidates : [];
        const lines = [`===== candidates | count=${list.length} =====`];
        list.forEach((candidate, index) => {
            lines.push('');
            lines.push(formatCandidateBlock(candidate, `candidate #${index + 1}`));
        });
        return lines.join('\n');
    }

    function buildSearchReconReportText() {
        const lines = [];
        if (reconState.reportLines.length) {
            lines.push(...reconState.reportLines);
        } else {
            const snapshot = scanSearchStatusNow();
            lines.push('Snov Search Status Recon Report');
            lines.push(`generatedAt: ${new Date().toLocaleString()}`);
            lines.push(`startUrl: ${location.href}`);
            lines.push(`title: ${document.title}`);
            lines.push('');
            lines.push(`result: ${snapshot.found ? 'found' : 'not_found'}`);
            lines.push(`exactMatch: ${snapshot.exactMatch ? 'yes' : 'no'}`);
            lines.push('');
            if (snapshot.best) {
                lines.push(formatCandidateBlock(snapshot.best, 'bestCandidate'));
                lines.push('');
            }
            lines.push(formatCandidateList(snapshot.candidates));
        }
        return lines.join('\n');
    }

    function downloadTextFile(filename, content) {
        const blob = new Blob([content], { type: 'text/plain;charset=utf-8' });
        const url = URL.createObjectURL(blob);
        const a = document.createElement('a');
        a.href = url;
        a.download = filename;
        document.body.appendChild(a);
        a.click();
        a.remove();
        setTimeout(() => URL.revokeObjectURL(url), 1000);
    }

    async function waitForEmailSearchComplete(maxWait = EMAIL_SEARCH_WAIT_TIMEOUT_MS) {
        const initialMatches = collectActiveEmailSearchingElements();

        if (!initialMatches.length) {
            logMessage('未检测到邮箱搜索状态，直接继续');
            return true;
        }

        logMessage(`检测到邮箱仍在搜索，当前命中 ${initialMatches.length} 处状态，等待完成...`);

        const start = Date.now();
        let lastLogAt = 0;
        let stableStartAt = null;

        while (Date.now() - start < maxWait) {
            ensureRunning();

            const matches = collectActiveEmailSearchingElements();
            if (!matches.length) {
                if (!stableStartAt) {
                    stableStartAt = Date.now();
                    logMessage(`搜索状态已全部消失，开始 ${EMAIL_SEARCH_STABLE_BUFFER_MS / 1000} 秒缓冲确认...`, 'success');
                }

                const stableElapsed = Date.now() - stableStartAt;
                if (stableElapsed >= EMAIL_SEARCH_STABLE_BUFFER_MS) {
                    logMessage('邮箱搜索已完成，继续导出流程', 'success');
                    return true;
                }

                await wait(300);
                continue;
            }

            stableStartAt = null;

            const now = Date.now();
            if (now - lastLogAt > 15000) {
                const elapsedSec = Math.floor((now - start) / 1000);
                const topCount = matches.filter(item => item.source === 'top-block').length;
                const rowCount = matches.filter(item => item.source === 'row-inline').length;
                logMessage(`邮箱仍在搜索，已等待 ${elapsedSec} 秒；顶部 ${topCount} 处，列表 ${rowCount} 处，总计 ${matches.length} 处...`);
                lastLogAt = now;
            }

            await wait(1000);
        }

        throw new Error('等待邮箱搜索完成超时（已超过 10 分钟）');
    }

    function getClickableParent(el, root) {
        let current = el;
        while (current && current !== root && current !== document.body) {
            if (
                current.tagName === 'BUTTON' ||
                current.tagName === 'A' ||
                current.getAttribute('role') === 'button' ||
                current.classList.contains('app-dropdown__drop-item') ||
                typeof current.onclick === 'function'
            ) {
                return current;
            }
            current = current.parentElement;
        }
        return el;
    }

    function findExportButtonInDropdown(dropdown) {
        if (!dropdown) return null;

        const candidates = [...dropdown.querySelectorAll('div, span, button, a, svg, use')];

        for (const el of candidates) {
            const text = getElementText(el);
            const localContainer = el.closest('div, button, a, span') || el;
            const hasExportIcon =
                !!localContainer.querySelector?.('use[xlink\\:href="#export_icon"]') ||
                !!el.querySelector?.('use[xlink\\:href="#export_icon"]') ||
                (el.tagName === 'USE' && el.getAttribute('xlink:href') === '#export_icon');

            if (text.includes('导出潜在客户') && hasExportIcon) {
                return getClickableParent(localContainer, dropdown);
            }
        }

        for (const el of candidates) {
            const text = getElementText(el);
            if (text.includes('导出潜在客户')) {
                const base = el.closest('div, button, a, span') || el;
                return getClickableParent(base, dropdown);
            }
        }

        const iconUse = dropdown.querySelector('use[xlink\\:href="#export_icon"]');
        if (iconUse) {
            const base = iconUse.closest('div, button, a, span') || iconUse;
            return getClickableParent(base, dropdown);
        }

        return null;
    }

    function findCommonAncestor(elements) {
        const valid = (elements || []).filter(Boolean);
        if (!valid.length) return null;
        if (valid.length === 1) return valid[0].parentElement || valid[0];

        const ancestorChains = valid.map(el => {
            const chain = [];
            let cur = el;
            while (cur && cur !== document.body) {
                chain.push(cur);
                cur = cur.parentElement;
            }
            chain.push(document.body);
            return chain;
        });

        for (const candidate of ancestorChains[0]) {
            if (ancestorChains.every(chain => chain.includes(candidate))) {
                return candidate;
            }
        }
        return null;
    }

    function findVisibleElementByText(texts, root = document) {
        const wanted = Array.isArray(texts) ? texts : [texts];
        const selectors = 'button, a, div, span';
        const nodes = root.querySelectorAll(selectors);

        for (const node of nodes) {
            if (!isVisible(node)) continue;
            const text = getElementText(node);
            if (!text) continue;
            if (wanted.some(t => text.includes(t))) {
                return node;
            }
        }
        return null;
    }

    function getProspectsToolbarRoot() {
        const addBtn = findVisibleElementByText('添加潜在客户');
        const sendBtn = findVisibleElementByText('发送营销邮件');
        const verifyBtn = findVisibleElementByText('验证邮箱地址');

        const common = findCommonAncestor([addBtn, sendBtn, verifyBtn].filter(Boolean));
        if (common && common !== document.body) {
            return common;
        }

        return addBtn?.parentElement || sendBtn?.parentElement || verifyBtn?.parentElement || null;
    }

    function getProspectsMenuButtonCandidates(root = null) {
        const scope = root || document;
        const candidates = [];
        const seen = new Set();

        const push = (el, scoreBase = 0) => {
            if (!el || seen.has(el) || !isVisible(el)) return;
            seen.add(el);

            const text = getElementText(el);
            const rect = el.getBoundingClientRect();
            let score = scoreBase;

            if (el.matches('button.prospects-context-menu__btn')) score += 100;
            if (el.matches('button.snv-btn--with-icon-only')) score += 80;
            if (el.getAttribute('aria-label')) score += 10;
            if (!text || text === '...' || text === '⋯') score += 20;
            if (rect.width <= 64) score += 15;
            if (rect.height <= 48) score += 10;
            if (el.querySelector('svg, use')) score += 10;

            candidates.push({ element: el, score, text, rect });
        };

        scope.querySelectorAll('button.prospects-context-menu__btn, button.snv-btn--with-icon-only').forEach(el => push(el, 0));
        scope.querySelectorAll('button').forEach(el => {
            const text = getElementText(el);
            if (!text || text === '...' || text === '⋯' || (el.querySelector('svg, use') && el.getBoundingClientRect().width <= 64)) {
                push(el, -10);
            }
        });

        candidates.sort((a, b) => {
            if (b.score !== a.score) return b.score - a.score;
            return a.rect.x - b.rect.x;
        });

        return candidates;
    }

        }

        await waitForCondition(() => {
            const nowLink = getCompaniesMenuLink();
            return !!(nowLink && nowLink.classList.contains('active')) || /\/companies(?:\/|$)/.test(location.pathname);
        }, 10000, 250);
        logMessage('已切换到【公司】界面', 'success');
        return true;
    }

    async function waitForExportCompletion(maxWait = EXPORT_SUCCESS_TIMEOUT_MS) {
        logMessage('等待导出成功提示，最长 2 分钟...');

        const start = Date.now();
        let lastLogAt = 0;
        let sawExporting = false;

        while (Date.now() - start < maxWait) {
            ensureRunning();

            const successAlert = getExportSuccessAlert();
            if (successAlert) {
                const downloadLink = successAlert.querySelector('a.btn-export, a[data-test="prospects-export-alert-no-google"]');
                if (downloadLink) {
                    logMessage('🎉 导出完成，可下载列表', 'success');
                } else {
                    logMessage('🎉 导出完成', 'success');
                }
                return true;
            }

            const exportingBlock = getExportingBlock();
            if (exportingBlock && !sawExporting) {
                sawExporting = true;
                logMessage('已检测到“正在导出”状态，继续等待导出成功...', 'success');
            } else if (exportingBlock) {
                sawExporting = true;
            }

            const now = Date.now();
            if (now - lastLogAt > 10000) {
                const elapsedSec = Math.floor((now - start) / 1000);
                if (exportingBlock) {
                    logMessage(`正在导出，已等待 ${elapsedSec} 秒...`);
                } else if (sawExporting) {
                    logMessage(`导出处理中，已等待 ${elapsedSec} 秒...`);
                } else {
                    logMessage(`等待导出成功提示，已等待 ${elapsedSec} 秒...`);
                }
                lastLogAt = now;
            }

            await wait(1000);
        }

        throw new Error('等待导出成功超时（已超过 2 分钟）');
    }

    async function runExportAutomation() {
        if (window.isExportRunning) {
            logMessage('导出已在运行中', 'warn');
            return;
        }

        window.isExportRunning = true;
        updateStatus('运行中', true);
        logMessage('🚀 开始自动导出流程...');

        try {
            ensureRunning();
            await switchToProspectsPageIfNeeded();

            ensureRunning();
            await waitForProspectsPageReady();

            ensureRunning();
            await waitForEmailSearchComplete();

            ensureRunning();
            await clickExportMenu();

            ensureRunning();
            await clickExportLeads();

            ensureRunning();
            const emailResult = await selectEmailStatusesInExportDialog();
            if (!emailResult) {
                logMessage('邮箱状态选择可能失败，继续执行...', 'error');
            }

            await wait(300);

            ensureRunning();
            const linkedinResult = await selectLinkedInProfilesInExportDialog();
            if (!linkedinResult) {
                logMessage('领英资料选择可能失败，继续执行...', 'error');
            }

            await wait(500);

            ensureRunning();
            await clickExportConfirm();

            ensureRunning();
            await waitForExportCompletion(EXPORT_SUCCESS_TIMEOUT_MS);

            ensureRunning();
            await switchToCompaniesPageAfterExport();

        } catch (error) {
            logMessage(`❌ 自动导出失败: ${error.message}`, 'error');
        } finally {
            window.isExportRunning = false;
            updateStatus('就绪', false);
        }
    }

    function stopExportAutomation() {
        window.isExportRunning = false;
        updateStatus('已停止', false);
        logMessage('导出已停止', 'warn');
    }

    async function testExportStep(step) {
        if (!window.isExportRunning) {
            window.isExportRunning = true;
            updateStatus('测试中', true);
        }

        try {
            logMessage(`测试步骤: ${step}`);

            switch (step) {
                case 'go-prospects':
                    return await switchToProspectsPageIfNeeded();
                case 'wait-email':
                    return await waitForEmailSearchComplete();
                case 'menu':
                    return await clickExportMenu();
                case 'export':
                    return await clickExportLeads();
                case 'email':
                    return await selectEmailStatusesInExportDialog();
                case 'linkedin':
                    return await selectLinkedInProfilesInExportDialog();
                case 'confirm':
                    return await clickExportConfirm();
                case 'wait-export':
                    return await waitForExportCompletion();
                case 'go-companies':
                    return await switchToCompaniesPageAfterExport();
                case 'relocate':
                    relocateExportButton();
                    return true;
                case 'detect-search':
                    return await runSearchStatusRecon();
                default:
                    logMessage('未知测试步骤', 'error');
                    return false;
            }
        } catch (e) {
            logMessage(`测试失败: ${e.message}`, 'error');
            return false;
        } finally {
            window.isExportRunning = false;
            updateStatus('就绪', false);
        }
    }
