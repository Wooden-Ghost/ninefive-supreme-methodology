  function nowTime() {
    const d = new Date();
    return [
      String(d.getHours()).padStart(2, '0'),
      String(d.getMinutes()).padStart(2, '0'),
      String(d.getSeconds()).padStart(2, '0'),
    ].join(':');
  }

    try { return root.querySelector(selector); } catch { return null; }

    if (!el) return '';

  function normalize(s) {
    return String(s || '')
      .toLowerCase()
      .replace(/\s+/g, '')
      .replace(/[._\-—–·•，,。/\\|()[\]{}【】"'“”‘’:：;；]/g, '');
  }

  function isVisible(el) {
    if (!el || !el.isConnected) return false;
    const rect = el.getBoundingClientRect();
    const style = getComputedStyle(el);
    return rect.width > 0 &&
      rect.height > 0 &&
      style.display !== 'none' &&
      style.visibility !== 'hidden' &&
      style.opacity !== '0';
  }

  function isEnabled(el) {
    if (!el) return false;
    return !el.disabled && !el.hasAttribute('disabled') && el.getAttribute('aria-disabled') !== 'true';
  }

  function isDisabledLike(el) {
    if (!el) return true;
    const cls = String(el.className || '');
    const ariaDisabled = el.getAttribute('aria-disabled');
    return Boolean(
      el.disabled ||
      ariaDisabled === 'true' ||
      cls.includes('disabled') ||
      cls.includes('snv-btn--disabled') ||
      cls.includes('n-button--disabled') ||
      el.getAttribute('disabled') !== null
    );
  }

  function isClickable(el) {
    if (!isVisible(el) || isDisabledLike(el)) return false;
    const rect = el.getBoundingClientRect();
    const x = Math.max(0, Math.min(window.innerWidth - 1, rect.left + rect.width / 2));
    const y = Math.max(0, Math.min(window.innerHeight - 1, rect.top + rect.height / 2));
    const topEl = document.elementFromPoint(x, y);
    return Boolean(topEl && (topEl === el || el.contains(topEl) || topEl.contains(el)));
  }

  function cssPath(el) {
    if (!el || !el.nodeType || el.nodeType !== 1) return '';
    const parts = [];
    let cur = el;

    while (cur && cur.nodeType === 1 && cur !== document.body && cur !== document.documentElement) {
      let part = cur.tagName.toLowerCase();

      if (cur.id) {
        part += '#' + CSS.escape(cur.id);
        parts.unshift(part);
        break;
      }

      const cls = Array.from(cur.classList || [])
        .filter(Boolean)
        .slice(0, 3)
        .map(c => '.' + CSS.escape(c))
        .join('');

      if (cls) part += cls;

      const parent = cur.parentElement;
      if (parent) {
        const sameTag = Array.from(parent.children).filter(x => x.tagName === cur.tagName);
        if (sameTag.length > 1) part += `:nth-of-type(${sameTag.indexOf(cur) + 1})`;
      }

      parts.unshift(part);
      cur = cur.parentElement;
      if (parts.length > 7) break;
    }

    return parts.join(' > ');
  }

  function buildStableSelector(el) {
    if (!el) return null;

    if (el.id) {
      return `#${CSS.escape(el.id)}`;
    }

    const dataTest = el.getAttribute('data-test');
    if (dataTest) {
      return `[data-test="${CSS.escape(dataTest)}"]`;
    }

    const trackingId = el.getAttribute('plerdy-tracking-id');
    if (trackingId) {
      return `[plerdy-tracking-id="${trackingId}"]`;
    }

    const classes = [...el.classList].filter(Boolean).slice(0, 4);
    if (classes.length) {
      return `${el.tagName.toLowerCase()}.${classes.map(c => CSS.escape(c)).join('.')}`;
    }

    return el.tagName.toLowerCase();
  }

  function getCssPath(el) {
    if (!el || el.nodeType !== 1) return '';
    const parts = [];
    let node = el;
    while (node && node.nodeType === 1 && node !== document.body) {
      let part = node.tagName.toLowerCase();
      if (node.id) {
        part += `#${node.id}`;
        parts.unshift(part);
        break;
      }
      const cls = [...node.classList].filter(Boolean).slice(0, 2);
      if (cls.length) {
        part += '.' + cls.join('.');
      }
      const parent = node.parentElement;
      if (parent) {
        const sameTagSiblings = [...parent.children].filter(child => child.tagName === node.tagName);
        if (sameTagSiblings.length > 1) {
          part += `:nth-of-type(${sameTagSiblings.indexOf(node) + 1})`;
        }
      }
      parts.unshift(part);
      node = node.parentElement;
    }
    return parts.join(' > ');
  }

  function limitText(text, maxLen = 260) {
    const s = String(text || '').replace(/\s+/g, ' ').trim();
    if (s.length <= maxLen) return s;
    return s.slice(0, maxLen) + '...';
  }

  function fireMouse(el, type) {
    if (!el) return;
    const rect = el.getBoundingClientRect();
    const event = new MouseEvent(type, {
      bubbles: true,
      cancelable: true,
      clientX: rect.left + rect.width / 2,
      clientY: rect.top + rect.height / 2,
    });
    el.dispatchEvent(event);
  }

  async function safeClickAsync(el, label = '元素') {
    if (!el) throw new Error(`未找到${label}`);
    if (!isVisible(el)) throw new Error(`${label}不可见`);
    if (isDisabledLike(el)) throw new Error(`${label}处于禁用状态`);

    el.scrollIntoView({ block: 'center', inline: 'center', behavior: 'instant' });
    await sleep(150);

    fireMouse(el, 'mouseover');
    fireMouse(el, 'mouseenter');
    fireMouse(el, 'mousemove');
    fireMouse(el, 'mousedown');
    fireMouse(el, 'mouseup');
    el.click();

    await sleep(300);
  }

  function safeClickSync(element) {
    if (!element) return false;

    try {
      element.scrollIntoView({ block: 'center', inline: 'center', behavior: 'auto' });
    } catch (e) {}

    try {
      element.dispatchEvent(new MouseEvent('mouseover', { bubbles: true, cancelable: true }));
      element.dispatchEvent(new MouseEvent('mousedown', { bubbles: true, cancelable: true }));
      element.dispatchEvent(new MouseEvent('mouseup', { bubbles: true, cancelable: true }));
      element.dispatchEvent(new MouseEvent('click', { bubbles: true, cancelable: true }));
      return true;
    } catch (e) {
      try {
        element.click();
        return true;
      } catch (e2) {
        return false;
      }
    }
  }

    if (APP.activeModule === module) renderLog();
  }

      line.textContent = `[${item.time}] ${item.message}`;
      box.appendChild(line);
    }

    box.scrollTop = box.scrollHeight;
  }

  function switchModule(target) {
    if (!['import', 'export'].includes(target)) return;
    if (APP.activeModule === target) return;

    APP.activeModule = target;
    APP.toolsVisible = false;

    const msg = target === 'import' ? '已切换至导入' : '已切换至导出';
    addLog('import', msg, 'ok');
    addLog('export', msg, 'ok');

    renderControls();
    renderStatus();
    renderLog();
    applyPanelModuleSize();
  }

    switchModule(APP.activeModule === 'import' ? 'export' : 'import');

  function bindDrag(panel) {
    const head = qs('#snu-drag', panel);
    if (!head) return;

    let dragging = false;
    let startX = 0;
    let startY = 0;
    let startRight = 0;
    let startBottom = 0;

    head.addEventListener('mousedown', (e) => {
      if (e.target.closest('button')) return;

      dragging = true;
      const rect = panel.getBoundingClientRect();

      startX = e.clientX;
      startY = e.clientY;
      startRight = window.innerWidth - rect.right;
      startBottom = window.innerHeight - rect.bottom;

      e.preventDefault();
    });

    window.addEventListener('mousemove', (e) => {
      if (!dragging) return;

      const dx = e.clientX - startX;
      const dy = e.clientY - startY;

      panel.style.right = Math.max(0, startRight - dx) + 'px';
      panel.style.bottom = Math.max(0, startBottom - dy) + 'px';
    });

    window.addEventListener('mouseup', () => {
      dragging = false;
    });
  }

  function toggleImportManualPause() {
    APP.importState.paused = !APP.importState.paused;

    if (APP.importState.paused) {
      setModuleStatus('import', '已暂停');
      addLog('import', '已暂停自动流程。搜索框为空时，再次点击【搜索】继续。', 'warn');
    } else {
      setModuleStatus('import', '自动检测中');
      addLog('import', '已继续自动流程。', 'ok');
      setTimeout(() => {
        importAutoTick();
      }, 80);
    }

  function importPauseCheckpoint(label = '') {
    if (APP.importState.closed) return false;

    if (APP.importState.paused) {
      setModuleStatus('import', '已暂停');
      if (label) {
        addLog('import', `已暂停，停止后续动作：${label}`, 'warn');
      }
      return false;
    }

    return true;
  }

  function readImportCycle() {
    try {
      const raw = sessionStorage.getItem(APP.importStorageKey);
      if (!raw) return null;
      return JSON.parse(raw);
    } catch {
      return null;
    }
  }

  function writeImportCycle(cycle) {
    sessionStorage.setItem(APP.importStorageKey, JSON.stringify(cycle));
  }

  function createImportCycle(reason, listName = '') {
    const title = getCurrentListTitle();
    const key = `${Date.now()}_${Math.random().toString(16).slice(2)}`;
    APP.importState.uploadRetryCount = 0;

    const cycle = {
      key,
      reason,
      listName: listName || title || '',
      createdAt: Date.now(),
      updatedAt: Date.now(),
      verifyBeforeImportDone: false,
      importDialogOpened: false,
      uploadClicked: false,
      nextClicked: false,
      importClicked: false,
      verifyAfterImportDone: false,
      completed: false,
      waitingFileSince: 0,
    };

    writeImportCycle(cycle);
    addLog('import', `已开启自动流程：${cycle.listName || reason}`, 'ok');
    return cycle;
  }

  function patchImportCycle(patch) {
    const cycle = readImportCycle() || createImportCycle('auto_bootstrap');
    const next = {
      ...cycle,
      ...patch,
      updatedAt: Date.now(),
    };
    writeImportCycle(next);
    return next;
  }

  function ensureImportCycle() {
    let cycle = readImportCycle();
    if (!cycle || cycle.completed) {
      cycle = createImportCycle('page_load');
    }
    return cycle;
  }

  function markImportCycleDone(reason) {
    const cycle = patchImportCycle({
      completed: true,
      completedAt: Date.now(),
      doneReason: reason || '',
    });
    addLog('import', `本轮自动流程结束：${cycle.listName || cycle.reason}`, 'ok');
  }

  function getAllCandidateListItems() {
    const selectors = [
      '.item.item--editable.item--folder',
      '.item.item--folder',
      '.item--folder',
      '.item',
    ];

    const seen = new Set();
    const items = [];

    for (const sel of selectors) {
      for (const el of qsa(sel)) {
        if (seen.has(el)) continue;
        seen.add(el);

        const nameNode = qs('.item__name', el);
        const visibleName = textOf(nameNode);
        const inputValue = valueOfInput(qs('input', el));
        const name = visibleName || inputValue || textOf(el);

        if (!name) continue;

        const rect = el.getBoundingClientRect();
        if (rect.width < 20 || rect.height < 10) continue;

        items.push({
          el,
          nameNode,
          name,
          visibleName,
          inputValue,
          selector: cssPath(el),
        });
      }
    }

    return items;
  }

  function findScrollableAncestors(el) {
    const arr = [];
    let cur = el ? el.parentElement : null;

    while (cur && cur !== document.body && cur !== document.documentElement) {
      const st = getComputedStyle(cur);
      const overflowY = st.overflowY;
      const canScroll = cur.scrollHeight > cur.clientHeight + 5;

      if (canScroll || ['auto', 'scroll', 'overlay'].includes(overflowY)) {
        arr.push({
          el: cur,
          scrollHeight: cur.scrollHeight,
          clientHeight: cur.clientHeight,
          scrollTop: cur.scrollTop,
          overflowY,
          rect: cur.getBoundingClientRect(),
          selector: cssPath(cur),
        });
      }

      cur = cur.parentElement;
    }

    return arr;
  }

  function findLeftListScrollContainer(itemEl = null) {
    const items = getAllCandidateListItems().filter(x => isVisible(x.el));
    const target = itemEl || (items[0] && items[0].el);

    if (target) {
      const ancestors = findScrollableAncestors(target)
        .filter(x => x.rect.left < window.innerWidth * 0.45)
        .sort((a, b) => {
          const aw = Math.abs(a.rect.width - 320);
          const bw = Math.abs(b.rect.width - 320);
          const ah = b.scrollHeight - b.clientHeight;
          const bh = a.scrollHeight - a.clientHeight;
          return aw - bw || ah - bh;
        });

      if (ancestors[0]) return ancestors[0].el;
    }

    const all = Array.from(document.querySelectorAll('div, aside, nav, section'));
    const candidates = all.map(el => {
      const rect = el.getBoundingClientRect();
      const itemCount = qsa('.item__name, .item--folder, .item', el).length;

      return {
        el,
        rect,
        canScroll: el.scrollHeight > el.clientHeight + 5,
        itemCount,
        score:
          (rect.left < window.innerWidth * 0.35 ? 20 : 0) +
          (el.scrollHeight > el.clientHeight + 5 ? 30 : 0) +
          Math.min(itemCount, 40) +
          (rect.height > 300 ? 10 : 0) -
          Math.abs(rect.width - 320) / 20,
      };
    }).filter(c =>
      c.rect.width > 120 &&
      c.rect.width < 540 &&
      c.rect.height > 200 &&
      c.itemCount > 2 &&
      c.rect.left < window.innerWidth * 0.45
    ).sort((a, b) => b.score - a.score);

    return candidates[0] ? candidates[0].el : null;
  }

  function highlightListItem(el) {
    qsa('.' + APP.highlightClass).forEach(x => x.classList.remove(APP.highlightClass));
    if (el) el.classList.add(APP.highlightClass);
  }

  function searchLastMatch(keywordRaw) {
    const keyword = normalize(keywordRaw);
    if (!keyword) return { keyword, matches: [], last: null };

    const items = getAllCandidateListItems();
    const matches = items.filter(item => {
      const visibleText = normalize(item.visibleName);
      const inputText = normalize(item.inputValue);
      const fullText = normalize(item.name);

      return visibleText.includes(keyword) || inputText.includes(keyword) || fullText.includes(keyword);
    });

    return {
      keyword,
      matches,
      last: matches.length ? matches[matches.length - 1] : null,
    };
  }

  async function centerItemInLeftContainer(item) {
    if (!item || !item.el) throw new Error('没有匹配项可定位');

    const container = findLeftListScrollContainer(item.el);

    if (!container) {
      item.el.scrollIntoView({ block: 'center', inline: 'nearest', behavior: 'smooth' });
      await sleep(500);
      return {
        container: null,
        centered: true,
      };
    }

    const beforeTop = container.scrollTop;
    const itemRect = item.el.getBoundingClientRect();
    const conRect = container.getBoundingClientRect();
    const targetTop = container.scrollTop + (itemRect.top - conRect.top) - (conRect.height / 2) + (itemRect.height / 2);

    container.scrollTo({
      top: Math.max(0, targetTop),
      behavior: 'smooth',
    });

    await sleep(650);

    const afterTop = container.scrollTop;
    const newItemRect = item.el.getBoundingClientRect();
    const newConRect = container.getBoundingClientRect();
    const itemCenter = newItemRect.top + newItemRect.height / 2;
    const conCenter = newConRect.top + newConRect.height / 2;
    const centered = Math.abs(itemCenter - conCenter) < Math.max(40, newConRect.height * 0.2);

    return {
      container,
      beforeTop,
      afterTop,
      centered,
      itemCenter,
      conCenter,
      containerSelector: cssPath(container),
    };
  }

  async function doImportSearchCenterAndClick() {
    const input = qs('#snu-import-keyword');
    const keyword = input ? input.value.trim() : '';

    if (input) input.blur();

    if (!keyword) {
      toggleImportManualPause();
      return;
    }

    if (APP.importState.paused) {
      APP.importState.paused = false;
      setModuleStatus('import', '自动检测中');
      addLog('import', '检测到关键词搜索，自动流程由暂停状态切换为继续。', 'ok');
    }

    setModuleStatus('import', '搜索中...');
    const result = searchLastMatch(keyword);

    addLog('import', `搜索关键词：${keyword}`);
    addLog('import', `匹配数量：${result.matches.length}`);

    if (!result.last) {
      addLog('import', `未找到匹配项：${keyword}`, 'warn');
      setModuleStatus('import', '无匹配');
      return;
    }

    addLog('import', `取最后一个匹配项：${result.last.name}`, 'ok');
    addLog('import', `全部匹配项：${result.matches.map(x => x.name).join(' ｜ ')}`);

    const info = await centerItemInLeftContainer(result.last);
    highlightListItem(result.last.el);

    if (info.container) {
      addLog('import', `已滚动到视觉中心；容器=${info.containerSelector}；是否居中=${info.centered ? '是' : '否'}`);
    } else {
      addLog('import', '未识别内部滚动容器，已使用 scrollIntoView 定位。', 'warn');
    }

    await sleep(250);

    if (!importPauseCheckpoint('点击搜索匹配列表前')) return;

    const clickTarget = getListClickTarget(result.last);
    addLog('import', `自动点击列表项：${result.last.name}`);
    await safeClickAsync(clickTarget, `列表项 ${result.last.name}`);

    createImportCycle('search_click', result.last.name);
    setModuleStatus('import', `已点击：${result.last.name}`);

    await waitForPageStable({ timeout: 20000, stableMs: 900, label: '列表切换后页面' });
  }

  async function waitForPageStable(options = {}) {
    const {
      timeout = 20000,
      stableMs = 900,
      label = '页面',
    } = options;

    setModuleStatus('import', `等待${label}稳定...`);

    const start = Date.now();
    let lastSig = '';
    let lastChange = Date.now();

    while (Date.now() - start < timeout) {
      const bodyText = document.body ? document.body.innerText : '';
      const sig = [
        document.readyState,
        location.href,
        bodyText.length,
        qsa('button').length,
        qsa('.refresh-img, [class*="loading"], [class*="spinner"], [class*="skeleton"], svg use[href*="loading"], svg use[xlink\\:href*="loading"]').length,
      ].join('|');

      if (sig !== lastSig) {
        lastSig = sig;
        lastChange = Date.now();
      }

      if (document.readyState === 'complete' && Date.now() - lastChange >= stableMs) {
        addLog('import', `${label}已稳定。`);
        setModuleStatus('import', `${label}已稳定`);
        return true;
      }

      await sleep(250);
    }

    addLog('import', `${label}等待稳定超时，继续尝试。`, 'warn');
    setModuleStatus('import', `${label}稳定超时`);
    return false;
  }

    });
  }

  async function waitForMainPageReady(timeout = 30000) {
    const start = Date.now();
    let lastLogAt = 0;

    while (Date.now() - start < timeout) {
      if (!importPauseCheckpoint('等待主页可操作')) return false;

      await waitForPageStable({ timeout: 5000, stableMs: 700, label: '主页面' });

      if (hasBlockingModal()) {
        if (Date.now() - lastLogAt > 5000) {
          addLog('import', '检测到可能阻挡操作的弹窗，请人工关闭后脚本会继续。', 'warn');
          lastLogAt = Date.now();
        }
        await sleep(1000);
        continue;
      }

      if (!looksLikeImportWizardPage() && (findAddProspectButton() || findVerifyEmailButton() || detectVerificationRunning().running)) {
        return true;
      }

      if (Date.now() - lastLogAt > 5000) {
        addLog('import', '等待主页面关键按钮出现：添加潜在客户 / 验证邮箱地址。', 'warn');
        lastLogAt = Date.now();
      }

      await sleep(800);
    }

    addLog('import', '等待主页面可操作超时，继续按当前页面尝试。', 'warn');
    return false;
  }

    APP.importState.lastVerifyClickAt = Date.now();
    await sleep(900);

    return 'clicked';
  }

  async function waitVerificationFinish(maxMs = 5 * 60 * 1000) {
    const start = Date.now();
    let logged = false;

    while (Date.now() - start < maxMs) {
      if (!importPauseCheckpoint('等待邮箱验证完成')) return false;

      const running = detectVerificationRunning();

      if (!running.running) {
        if (logged) addLog('import', '验证状态已结束。', 'ok');
        return true;
      }

      if (!logged) {
        addLog('import', `等待邮箱验证完成：${running.texts.join(' ｜ ') || '验证中'}`, 'warn');
        logged = true;
      }

      setModuleStatus('import', '邮箱验证中...');
      await sleep(3000);
    }

    addLog('import', '等待邮箱验证完成超时，本轮停止，避免误操作。', 'err');
    return false;
  }

        const titleHit = /从\s*文件\s*导入/i.test(t);
        const uploadHit = /拖放\s*文件.*上传|选择\s*文件|上传\s*文件|Excel、?\s*CSV\s*或\s*TXT|Excel,\s*CSV|TXT\s*文件/i.test(t);

        return { el, r, t, area, centered, titleHit, uploadHit };
      })
      .filter(x =>
        x.titleHit &&
        x.uploadHit &&
        x.centered &&
        x.r.width >= 280 &&
        x.r.width <= 900 &&
        x.r.height >= 180 &&
        x.r.height <= 760
      )
      .sort((a, b) => a.area - b.area);

    return candidates[0] ? candidates[0].el : null;
  }

  async function handleFileImportModal(cycle) {
    if (looksLikeImportWizardPage() && /字段映射|数据设置|下一步|导入/.test(textOf(document.body))) {
      return false;
    }

    if (!isFileImportModalOpen()) return false;

    if (!importPauseCheckpoint('文件导入弹窗处理中')) return true;

    const upload = findUploadFileButton();
    const selected = modalHasSelectedFile();

    if (upload && isVisible(upload) && !isDisabledLike(upload)) {
      const elapsed = Date.now() - APP.importState.lastUploadClickAt;

      if (cycle.uploadClicked && elapsed < 12000) {
        setModuleStatus('import', '上传已点击，等待跳转');
        return true;
      }

      if (cycle.uploadClicked && elapsed >= 12000) {
        if (APP.importState.uploadRetryCount >= APP.importState.uploadRetryLimit) {
          addLog('import', '上传后仍停留在文件导入弹窗，已达到重试上限。请手动点击“上传文件”或重新选择文件。', 'err');
          setModuleStatus('import', '上传未跳转，需人工处理');
          return true;
        }

        APP.importState.uploadRetryCount += 1;
        addLog('import', `上传后仍停留在弹窗，准备重试上传：${APP.importState.uploadRetryCount}/${APP.importState.uploadRetryLimit}`, 'warn');
        patchImportCycle({ uploadClicked: false });
      }

      if (!cycle.uploadClicked || elapsed >= 12000) {
        if (!importPauseCheckpoint('点击上传文件前')) return true;
        APP.importState.lastUploadClickAt = Date.now();

        addLog('import', '检测到“上传文件”按钮可点击，自动上传。', 'ok');
        await safeClickAsync(upload, '上传文件按钮');
        addLog('import', '已点击“上传文件”。', 'ok');

        patchImportCycle({ uploadClicked: true });

        await sleep(1200);
        if (!importPauseCheckpoint('上传文件后等待跳转前')) return true;
        await waitForPageStable({ timeout: 15000, stableMs: 900, label: '上传后页面' });

        return true;
      }
    }

    if (upload && isDisabledLike(upload)) {
      setModuleStatus('import', '等待上传按钮可点');
      addLog('import', '“上传文件”按钮存在但不可点击，继续等待。', 'warn');
      return true;
    }

    if (!selected) {
      if (!cycle.waitingFileSince) {
        patchImportCycle({ waitingFileSince: Date.now() });
      }

      setModuleStatus('import', '等待人工选择文件');
      return true;
    }

    addLog('import', '检测到文件已选择，但暂未找到可点击的“上传文件”按钮，继续等待。', 'warn');
    setModuleStatus('import', '等待上传文件按钮');
    return true;
  }

  function currentWizardStep() {
    const text = textOf(document.body);
    const next = findNextButton();
    const imp = findImportButton();

    if (next && !imp) return 'mapping';
    if (imp && !next) return 'settings';

    if (next && imp) {
      const nr = next.getBoundingClientRect();
      const ir = imp.getBoundingClientRect();
      return (ir.right + ir.bottom) > (nr.right + nr.bottom) ? 'settings' : 'mapping';
    }

    if (/重复设置|选择\s*要添加潜在客户的列表|导入位置|合并数据|创建多个记录|不上传/.test(text)) {
      return 'settings';
    }

    if (/拖放\s*字段|跳过第一行|添加自定义字段/.test(text)) {
      return 'mapping';
    }

    return 'unknown';
  }

  async function handleImportWizard(cycle) {
    if (!looksLikeImportWizardPage()) return false;

    if (!importPauseCheckpoint('导入向导页处理中')) return true;

    APP.importState.uploadRetryCount = 0;

    const step = currentWizardStep();
    setModuleStatus('import', `导入页：${step}`);
    addLog('import', `导入向导当前步骤判定：${step}`);

    if (step === 'mapping') {
      const nextBtn = findNextButton();

      if (!nextBtn) {
        addLog('import', '字段映射页：未找到“下一步”按钮。', 'err');
        return true;
      }

      addLog('import', `字段映射页：检测到下一步按钮；文本=${textOf(nextBtn)}；禁用=${isDisabledLike(nextBtn) ? '是' : '否'}；可见=${isVisible(nextBtn) ? '是' : '否'}`);

      if (isDisabledLike(nextBtn)) {
        addLog('import', '字段映射页：“下一步”不可点击，等待字段映射完成。', 'warn');
        return true;
      }

      if (Date.now() - APP.importState.lastNextClickAt < 5000) {
        return true;
      }

      if (cycle.nextClicked) {
        addLog('import', '字段映射页仍停留，但本轮记录已点过“下一步”，执行重试点击。', 'warn');
      }

      if (!importPauseCheckpoint('点击下一步前')) return true;
      APP.importState.lastNextClickAt = Date.now();

      await safeClickAsync(nextBtn, '下一步按钮');
      addLog('import', '已点击“下一步”。', 'ok');

      patchImportCycle({ nextClicked: true });
      await sleep(1200);
      if (!importPauseCheckpoint('点击下一步后等待数据设置页前')) return true;
      await waitForPageStable({ timeout: 20000, stableMs: 900, label: '数据设置页面' });

      return true;
    }

    if (step === 'settings') {
      const importBtn = findImportButton();

      if (!importBtn) {
        addLog('import', '数据设置页：未找到“导入”按钮。', 'err');
        return true;
      }

      addLog('import', `数据设置页：检测到导入按钮；文本=${textOf(importBtn)}；禁用=${isDisabledLike(importBtn) ? '是' : '否'}；可见=${isVisible(importBtn) ? '是' : '否'}`);

      if (isDisabledLike(importBtn)) {
        addLog('import', '数据设置页：“导入”按钮不可点击，等待页面设置完成。', 'warn');
        return true;
      }

      if (Date.now() - APP.importState.lastImportClickAt < 5000) {
        return true;
      }

      if (cycle.importClicked) {
        addLog('import', '数据设置页仍停留，但本轮记录已点过“导入”，执行重试点击。', 'warn');
      }

      if (!importPauseCheckpoint('点击导入前')) return true;
      APP.importState.lastImportClickAt = Date.now();

      await safeClickAsync(importBtn, '导入按钮');
      addLog('import', '已点击“导入”，等待返回主列表页。', 'ok');

      patchImportCycle({ importClicked: true });
      await sleep(1200);
      if (!importPauseCheckpoint('点击导入后等待主列表页前')) return true;
      await waitForPageStable({ timeout: 45000, stableMs: 1200, label: '导入后页面' });

      return true;
    }

    addLog('import', '导入页面步骤未知，等待页面稳定或人工确认。', 'warn');
    return true;
  }

  async function handleMainPage(cycle) {
    if (!looksLikeMainProspectsPage()) return false;

    if (!importPauseCheckpoint('主页处理中')) return true;

    await waitForMainPageReady();

    if (!importPauseCheckpoint('主页准备完成后')) return true;

    const running = detectVerificationRunning();

    if (running.running) {
      await waitVerificationFinish();
      return true;
    }

    const verifyBtn = findVerifyEmailButton();
    const verifyClickable = Boolean(
      verifyBtn &&
      isVisible(verifyBtn) &&
      !isDisabledLike(verifyBtn) &&
      Date.now() - APP.importState.lastVerifyClickAt > 6000
    );

    if (verifyClickable) {
      const res = await clickVerifyIfClickable(cycle.importClicked ? '导入后检测' : '导入前检测');

      if (res === 'clicked') {
        if (cycle.importClicked) {
          patchImportCycle({ verifyAfterImportDone: true });
        } else {
          patchImportCycle({ verifyBeforeImportDone: true });
        }

        await waitVerificationFinish();

        if (cycle.importClicked) {
          markImportCycleDone('导入后验证已执行');
          setModuleStatus('import', '本轮完成');
        }

        return true;
      }

      if (res === 'running') {
        await waitVerificationFinish();
        return true;
      }

      if (res === 'paused') return true;
    }

    if (!cycle.verifyBeforeImportDone && !cycle.importClicked) {
      addLog('import', '导入前没有可点击的验证按钮，继续打开文件导入。');
      patchImportCycle({ verifyBeforeImportDone: true });
    }

    if (!cycle.importClicked) {
      setModuleStatus('import', '准备从文件导入');
      await openImportDialogIfNeeded();
      return true;
    }

    if (cycle.importClicked && !cycle.verifyAfterImportDone) {
      patchImportCycle({ verifyAfterImportDone: true });
      markImportCycleDone('导入后无需验证或按钮不可用');
      setModuleStatus('import', '本轮完成');
      return true;
    }

    return true;
  }

  async function importAutoTick() {
    if (APP.importState.closed || APP.importState.locked) return;

    /*
     * 合并后，只有当前模块是“导入”时才运行导入自动流程。
     * 避免切到导出模块后，导入逻辑继续自动弹出文件导入窗口。
     */
    if (APP.activeModule !== 'import') return;

    if (APP.importState.paused) {
      setModuleStatus('import', '已暂停');
      return;
    }

    if (isSearchBoxFocused()) {
      setModuleStatus('import', '搜索框输入中，自动暂停');
      if (Date.now() - APP.importState.lastSearchPauseLogAt > 6000) {
        addLog('import', '搜索框获得焦点，自动流程临时暂停；输入关键词后按 Enter / 点搜索会执行搜索。', 'warn');
        APP.importState.lastSearchPauseLogAt = Date.now();
      }
      return;
    }

    APP.importState.locked = true;

    try {
      const cycle = ensureImportCycle();

      if (cycle.completed) {
        setModuleStatus('import', '待命');
        return;
      }

      if (Date.now() - APP.importState.lastActionAt < 500) {
        return;
      }

      const wizardHandled = await handleImportWizard(cycle);
      if (wizardHandled) return;

      const modalHandled = await handleFileImportModal(cycle);
      if (modalHandled) return;

      const mainHandled = await handleMainPage(cycle);
      if (mainHandled) return;

      setModuleStatus('import', '未识别页面，等待');
    } catch (err) {
      addLog('import', `自动流程异常：${err.message || err}`, 'err');
      setModuleStatus('import', '自动流程异常');
    } finally {
      APP.importState.lastActionAt = Date.now();
      APP.importState.locked = false;
    }
  }

  function waitForExportCondition(fn, timeout = 10000, interval = 250) {
    return new Promise((resolve, reject) => {
      const start = Date.now();

      const check = () => {
        if (!APP.exportState.running) {
          reject(new Error('任务已停止'));
          return;
        }

        try {
          const result = fn();
          if (result) {
            resolve(result);
            return;
          }
        } catch (e) {}

        if (Date.now() - start > timeout) {
          reject(new Error('等待超时'));
          return;
        }

        setTimeout(check, interval);
      };

      check();
    });
  }

  function getEmailSearchingBlock() {
    const blocks = document.querySelectorAll('.check-verification-block');

    for (const block of blocks) {
      if (!isVisible(block)) continue;

      const text = (block.textContent || '').trim();
      const hasAnimatedLoadingIcon = !!block.querySelector('use[xlink\\:href="#loading_icon__animated"]');
      const hasSearchingText = text.includes('正在搜索邮箱地址');

      if (hasAnimatedLoadingIcon && hasSearchingText) {
        return block;
      }
    }

    return null;
  }

  function collectActiveEmailSearchingElements() {
    const matches = [];
    const seen = new Set();

    function pushMatch(el, source) {
      if (!el || seen.has(el) || !isVisible(el)) return;
      const text = getElementText(el);
      if (!text.includes('正在搜索邮箱地址')) return;
      seen.add(el);
      matches.push({
        element: el,
        source,
        text,
        selector: buildStableSelectorSafe(el),
        cssPath: getCssPath(el),
      });
    }

    document.querySelectorAll('.check-verification-block').forEach(block => {
      const text = getElementText(block);
      const hasAnimatedLoadingIcon = !!block.querySelector('use[xlink\\:href="#loading_icon__animated"]');
      if (hasAnimatedLoadingIcon && text.includes('正在搜索邮箱地址')) {
        pushMatch(block, 'top-block');
      }
    });

    document.querySelectorAll('.generic__searching, .stop-drag, .row-info, td.row__cell--relative').forEach(el => {
      const text = getElementText(el);
      if (!text.includes('正在搜索邮箱地址')) return;

      const searchingNode = el.classList.contains('generic__searching')
        ? el
        : (el.querySelector('.generic__searching') || el);

      pushMatch(searchingNode, 'row-inline');
    });

    document.querySelectorAll('div, span').forEach(el => {
      if (seen.has(el) || !isVisible(el)) return;
      const text = getElementText(el);
      if (!text.includes('正在搜索邮箱地址')) return;
      if (text.length > 80) return;
      pushMatch(el, 'text-fallback');
    });

    return matches;
  }

  async function waitForEmailSearchComplete(maxWait = EXPORT.EMAIL_SEARCH_WAIT_TIMEOUT_MS) {
    const initialMatches = collectActiveEmailSearchingElements();

    if (!initialMatches.length) {
      exportLog('未检测到邮箱搜索状态，直接继续');
      return true;
    }

    exportLog(`检测到邮箱仍在搜索，当前命中 ${initialMatches.length} 处状态，等待完成...`);

    const start = Date.now();
    let lastLogAt = 0;
    let stableStartAt = null;

    while (Date.now() - start < maxWait) {
      ensureExportRunning();

      const matches = collectActiveEmailSearchingElements();
      if (!matches.length) {
        if (!stableStartAt) {
          stableStartAt = Date.now();
          exportLog(`搜索状态已全部消失，开始 ${EXPORT.EMAIL_SEARCH_STABLE_BUFFER_MS / 1000} 秒缓冲确认...`, 'ok');
        }

        const stableElapsed = Date.now() - stableStartAt;
        if (stableElapsed >= EXPORT.EMAIL_SEARCH_STABLE_BUFFER_MS) {
          exportLog('邮箱搜索已完成，继续导出流程', 'ok');
          return true;
        }

        await sleep(300);
        continue;
      }

      stableStartAt = null;

      const now = Date.now();
      if (now - lastLogAt > 15000) {
        const elapsedSec = Math.floor((now - start) / 1000);
        const topCount = matches.filter(item => item.source === 'top-block').length;
        const rowCount = matches.filter(item => item.source === 'row-inline').length;
        exportLog(`邮箱仍在搜索，已等待 ${elapsedSec} 秒；顶部 ${topCount} 处，列表 ${rowCount} 处，总计 ${matches.length} 处...`);
        lastLogAt = now;
      }

      await sleep(1000);
    }

    throw new Error('等待邮箱搜索完成超时（已超过 10 分钟）');
  }

  function buildStableSelectorSafe(el) {
    if (!el) return null;

    if (el.id) {
      return `#${CSS.escape(el.id)}`;
    }

    const dataTest = el.getAttribute('data-test');
    if (dataTest) {
      return `[data-test="${CSS.escape(dataTest)}"]`;
    }

    const trackingId = el.getAttribute('plerdy-tracking-id');
    if (trackingId) {
      return `[plerdy-tracking-id="${trackingId}"]`;
    }

    const classes = [...el.classList].filter(Boolean).slice(0, 4);
    if (classes.length) {
      return `${el.tagName.toLowerCase()}.${classes.map(c => CSS.escape(c)).join('.')}`;
    }

    return el.tagName.toLowerCase();
  }

  function getClickableParent(el, root) {
    let current = el;
    while (current && current !== root && current !== document.body) {
      if (
        current.tagName === 'BUTTON' ||
        current.tagName === 'A' ||
        current.getAttribute('role') === 'button' ||
        current.classList.contains('app-dropdown__drop-item') ||
        typeof current.onclick === 'function'
      ) {
        return current;
      }
      current = current.parentElement;
    }
    return el;
  }

  function findExportButtonInDropdown(dropdown) {
    if (!dropdown) return null;

    const candidates = [...dropdown.querySelectorAll('div, span, button, a, svg, use')];

    for (const el of candidates) {
      const text = getElementText(el);
      const localContainer = el.closest('div, button, a, span') || el;
      const hasExportIcon =
        !!localContainer.querySelector?.('use[xlink\\:href="#export_icon"]') ||
        !!el.querySelector?.('use[xlink\\:href="#export_icon"]') ||
        (el.tagName === 'USE' && el.getAttribute('xlink:href') === '#export_icon');

      if (text.includes('导出潜在客户') && hasExportIcon) {
        return getClickableParent(localContainer, dropdown);
      }
    }

    for (const el of candidates) {
      const text = getElementText(el);
      if (text.includes('导出潜在客户')) {
        const base = el.closest('div, button, a, span') || el;
        return getClickableParent(base, dropdown);
      }
    }

    const iconUse = dropdown.querySelector('use[xlink\\:href="#export_icon"]');
    if (iconUse) {
      const base = iconUse.closest('div, button, a, span') || iconUse;
      return getClickableParent(base, dropdown);
    }

    return null;
  }

  function findCommonAncestor(elements) {
    const valid = (elements || []).filter(Boolean);
    if (!valid.length) return null;
    if (valid.length === 1) return valid[0].parentElement || valid[0];

    const ancestorChains = valid.map(el => {
      const chain = [];
      let cur = el;
      while (cur && cur !== document.body) {
        chain.push(cur);
        cur = cur.parentElement;
      }
      chain.push(document.body);
      return chain;
    });

    for (const candidate of ancestorChains[0]) {
      if (ancestorChains.every(chain => chain.includes(candidate))) {
        return candidate;
      }
    }
    return null;
  }

  function findVisibleElementByText(texts, root = document) {
    const wanted = Array.isArray(texts) ? texts : [texts];
    const selectors = 'button, a, div, span';
    const nodes = root.querySelectorAll(selectors);

    for (const node of nodes) {
      if (!isVisible(node)) continue;
      const text = getElementText(node);
      if (!text) continue;
      if (wanted.some(t => text.includes(t))) {
        return node;
      }
    }
    return null;
  }

  function getProspectsToolbarRoot() {
    const addBtn = findVisibleElementByText('添加潜在客户');
    const sendBtn = findVisibleElementByText('发送营销邮件');
    const verifyBtn = findVisibleElementByText('验证邮箱地址');

    const common = findCommonAncestor([addBtn, sendBtn, verifyBtn].filter(Boolean));
    if (common && common !== document.body) {
      return common;
    }

    return addBtn?.parentElement || sendBtn?.parentElement || verifyBtn?.parentElement || null;
  }

  function getProspectsMenuButtonCandidates(root = null) {
    const scope = root || document;
    const candidates = [];
    const seen = new Set();

    const push = (el, scoreBase = 0) => {
      if (!el || seen.has(el) || !isVisible(el)) return;
      seen.add(el);

      const text = getElementText(el);
      const rect = el.getBoundingClientRect();
      let score = scoreBase;

      if (el.matches('button.prospects-context-menu__btn')) score += 100;
      if (el.matches('button.snv-btn--with-icon-only')) score += 80;
      if (el.getAttribute('aria-label')) score += 10;
      if (!text || text === '...' || text === '⋯') score += 20;
      if (rect.width <= 64) score += 15;
      if (rect.height <= 48) score += 10;
      if (el.querySelector('svg, use')) score += 10;

      candidates.push({ element: el, score, text, rect });
    };

    scope.querySelectorAll('button.prospects-context-menu__btn, button.snv-btn--with-icon-only').forEach(el => push(el, 0));
    scope.querySelectorAll('button').forEach(el => {
      const text = getElementText(el);
      if (!text || text === '...' || text === '⋯' || (el.querySelector('svg, use') && el.getBoundingClientRect().width <= 64)) {
        push(el, -10);
      }
    });

    candidates.sort((a, b) => {
      if (b.score !== a.score) return b.score - a.score;
      return a.rect.x - b.rect.x;
    });

    return candidates;
  }

  function getPreferredExportMenuButton() {
    const toolbar = getProspectsToolbarRoot();
    const scoped = toolbar ? getProspectsMenuButtonCandidates(toolbar) : [];
    if (scoped.length) return scoped[0].element;

    const global = getProspectsMenuButtonCandidates(document);
    return global.length ? global[0].element : null;
  }

  function getProspectsPageReadySnapshot() {
    const prospectsLink = getProspectsMenuLink();
    const pathOk = /\/prospects(?:\/|$)/.test(location.pathname);
    const menuActive = !!(prospectsLink && prospectsLink.classList.contains('active'));
    const toolbar = getProspectsToolbarRoot();
    const menuButton = getPreferredExportMenuButton();

    return {
      pathOk,
      menuActive,
      toolbarReady: !!toolbar,
      menuButtonReady: !!menuButton,
      toolbar,
      menuButton,
    };
  }

  async function waitForProspectsPageReady(timeout = EXPORT.PROSPECTS_PAGE_READY_TIMEOUT_MS) {
    exportLog('已切回潜在客户，等待页面刷新完成...');
    setModuleStatus('export', '等待页面稳定...');

    const start = Date.now();
    let stableSince = 0;
    let lastLogAt = 0;

    while (Date.now() - start < timeout) {
      ensureExportRunning();

      const snapshot = getProspectsPageReadySnapshot();
      const ready = snapshot.pathOk && snapshot.menuActive && snapshot.toolbarReady && snapshot.menuButtonReady;

      if (ready) {
        if (!stableSince) {
          stableSince = Date.now();
        }

        if (Date.now() - stableSince >= EXPORT.PROSPECTS_PAGE_STABLE_BUFFER_MS) {
          exportLog('潜在客户页面已稳定，可继续导出', 'ok');
          setModuleStatus('export', '页面已稳定');
          return true;
        }
      } else {
        stableSince = 0;
      }

      const now = Date.now();
      if (now - lastLogAt > 2000) {
        exportLog(
          `等待潜在客户页稳定：路径=${snapshot.pathOk ? '1' : '0'}，菜单=${snapshot.menuActive ? '1' : '0'}，工具栏=${snapshot.toolbarReady ? '1' : '0'}，导出菜单=${snapshot.menuButtonReady ? '1' : '0'}`
        );
        lastLogAt = now;
      }

      await sleep(250);
    }

    throw new Error('切回潜在客户后等待页面稳定超时');
  }

  async function switchToProspectsPageIfNeeded() {
    const link = getProspectsMenuLink();
    if (!link) {
      exportLog('未找到左侧【潜在客户】菜单，继续按当前页面执行', 'warn');
      return false;
    }

    const isActive = link.classList.contains('active') || /\/prospects(?:\/|$)/.test(location.pathname);
    if (isActive) {
      exportLog('当前已在潜在客户界面');
      return true;
    }

    exportLog('开始前切回【潜在客户】界面...');
    if (!safeClickSync(link)) {
      throw new Error('点击左侧【潜在客户】菜单失败');
    }

    await waitForExportCondition(() => {
      const nowLink = getProspectsMenuLink();
      return !!(nowLink && nowLink.classList.contains('active')) || /\/prospects(?:\/|$)/.test(location.pathname);
    }, 10000, 250);

    await sleep(300);
    exportLog('已切回【潜在客户】界面', 'ok');
    return true;
  }

  async function waitForExportCompletion(maxWait = EXPORT.EXPORT_SUCCESS_TIMEOUT_MS) {
    exportLog('等待导出成功提示，最长 2 分钟...');

    const start = Date.now();
    let lastLogAt = 0;
    let sawExporting = false;

    while (Date.now() - start < maxWait) {
      ensureExportRunning();

      const successAlert = getExportSuccessAlert();
      if (successAlert) {
        const downloadLink = successAlert.querySelector('a.btn-export, a[data-test="prospects-export-alert-no-google"]');
        if (downloadLink) {
          exportLog('🎉 导出完成，可下载列表', 'ok');
        } else {
          exportLog('🎉 导出完成', 'ok');
        }
        return true;
      }

      const exportingBlock = getExportingBlock();
      if (exportingBlock && !sawExporting) {
        sawExporting = true;
        exportLog('已检测到“正在导出”状态，继续等待导出成功...', 'ok');
      } else if (exportingBlock) {
        sawExporting = true;
      }

      const now = Date.now();
      if (now - lastLogAt > 10000) {
        const elapsedSec = Math.floor((now - start) / 1000);
        if (exportingBlock) {
          exportLog(`正在导出，已等待 ${elapsedSec} 秒...`);
        } else if (sawExporting) {
          exportLog(`导出处理中，已等待 ${elapsedSec} 秒...`);
        } else {
          exportLog(`等待导出成功提示，已等待 ${elapsedSec} 秒...`);
        }
        lastLogAt = now;
      }

      await sleep(1000);
    }

    throw new Error('等待导出成功超时（已超过 2 分钟）');
  }

  async function runExportAutomation() {
    if (APP.exportState.running) {
      exportLog('导出已在运行中', 'warn');
      return;
    }

    APP.exportState.running = true;
    updateExportStatus('运行中', true);
    exportLog('🚀 开始自动导出流程...');

    try {
      ensureExportRunning();
      await switchToProspectsPageIfNeeded();

      ensureExportRunning();
      await waitForProspectsPageReady();

      ensureExportRunning();
      await waitForEmailSearchComplete();

      ensureExportRunning();
      await clickExportMenu();

      ensureExportRunning();
      await clickExportLeads();

      ensureExportRunning();
      const emailResult = await selectEmailStatusesInExportDialog();
      if (!emailResult) {
        exportLog('邮箱状态选择可能失败，继续执行...', 'err');
      }

      await sleep(300);

      ensureExportRunning();
      const linkedinResult = await selectLinkedInProfilesInExportDialog();
      if (!linkedinResult) {
        exportLog('领英资料选择可能失败，继续执行...', 'err');
      }

      await sleep(500);

      ensureExportRunning();
      await clickExportConfirm();

      ensureExportRunning();
      await waitForExportCompletion(EXPORT.EXPORT_SUCCESS_TIMEOUT_MS);

      ensureExportRunning();
      await switchToCompaniesPageAfterExport();
    } catch (error) {
      exportLog(`❌ 自动导出失败: ${error.message}`, 'err');
    } finally {
      APP.exportState.running = false;
      updateExportStatus('就绪', false);
    }
  }

  function stopExportAutomation() {
    APP.exportState.running = false;
    updateExportStatus('已停止', false);
    exportLog('导出已停止', 'warn');
  }

  function setupGlobals() {
    window.snUnifiedHelper = {
      version: APP.version,
      switchModule,
      importPause: () => {
        if (!APP.importState.paused) toggleImportManualPause();
      },
      importResume: () => {
        if (APP.importState.paused) toggleImportManualPause();
      },
      exportRun: runExportAutomation,
      exportStop: stopExportAutomation,
      relocateExportButton,
    };
  }

  function start() {
    createPanel();
    setupGlobals();

    waitForPageStable({ timeout: 12000, stableMs: 900, label: '初始页面' })
      .then(() => {
        ensureImportCycle();
        setInterval(importAutoTick, APP.tickMs);
        importAutoTick();
      })
      .catch(() => {
        ensureImportCycle();
        setInterval(importAutoTick, APP.tickMs);
        importAutoTick();
      });
  }

  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', () => setTimeout(start, 800));
  } else {
    setTimeout(start, 800);
  }
})();
