/*
更新说明 v1.2.1
1. 移除“进入链接”“判断状态”“复制链接”按钮及对应入口功能，避免阶段边界混乱。
2. “开始爬取”只允许在“从收藏夹进入的视频播放页”运行；普通推荐页、搜索页、TA作品页、普通视频详情页均会拒绝爬取。
3. 状态区改为显示当前界面与是否可开始爬取，不再显示容易误导的通用“已完成加载”。
4. 保留 v1.1.0 已确认较满意的爬取与 CSV 导出逻辑：读取当前播放视频 ID + 当前详情区标题/文案，通过 ArrowDown/滚轮/右侧卡片备用动作切换，确认 ID 变化后才记录。
5. 清除缓存只清空结果、计数和运行状态，不清空日志；采集完成后不自动导出 CSV，需要人工点击【导出CSV】。

更新说明 v1.2.1
1. 修复旧版本面板关闭后定时刷新仍可能干扰新面板的问题：使用新 UI DOM ID，并加入单实例运行时销毁逻辑。
2. 关闭面板会销毁本实例的状态刷新定时器、拖拽监听、待执行 timeout 和运行状态。
3. 状态区拆分为页面状态与脚本运行状态；“暂停”只由用户点击暂停按钮触发，不再由播放器卡顿/加载状态推导。
4. 状态刷新改为节流刷新，且只有文本变化时才写 DOM，减少视频加载卡顿时的闪烁。
*/

    if (!el || isInPanel(el)) return false;

    const s = window.getComputedStyle(el);
    return r.width > 0 && r.height > 0 && s.display !== 'none' && s.visibility !== 'hidden' && s.opacity !== '0';

  function clickable(el) {
    if (!visible(el)) return false;
    const tag = (el.tagName || '').toLowerCase();
    return tag === 'a' || tag === 'button' || el.getAttribute('role') === 'button' || el.getAttribute('role') === 'tab' || window.getComputedStyle(el).cursor === 'pointer' || !!el.onclick;
  }

  function cssPath(el) {
    if (!el) return '';
    if (el.id) return `${el.tagName.toLowerCase()}#${CSS.escape(el.id)}`;
    const parts = [];
    let cur = el;
    for (let i = 0; cur && cur.nodeType === 1 && i < 5; i++, cur = cur.parentElement) {
      let part = cur.tagName.toLowerCase();
      if (cur.id) {
        part += '#' + CSS.escape(cur.id);
        parts.unshift(part);
        break;
      }
      const cls = [...cur.classList].filter(Boolean).slice(0, 3).map((c) => '.' + CSS.escape(c)).join('');
      if (cls) part += cls;
      parts.unshift(part);
    }
    return parts.join(' > ');
  }

  function pickVisible(selector, root = document) {
    try { return [...root.querySelectorAll(selector)].find(visible) || null; } catch { return null; }
  }

  function pickAllVisible(selector, root = document) {
    try { return [...root.querySelectorAll(selector)].filter(visible); } catch { return []; }
  }

  function centerOf(el) {
    const r = el.getBoundingClientRect();
    return { x: Math.round(r.left + r.width / 2), y: Math.round(r.top + r.height / 2), rect: r };
  }

  function mouseClick(el, label = '') {
    if (!el) return false;
    try { el.scrollIntoView({ block: 'center', inline: 'center' }); } catch {}
    const { x, y } = centerOf(el);
    const target = document.elementFromPoint(x, y) || el;
    const opts = { bubbles: true, cancelable: true, view: window, clientX: x, clientY: y, button: 0 };
    ['pointerdown', 'mousedown', 'pointerup', 'mouseup', 'click'].forEach((type) => {
      try { target.dispatchEvent(new MouseEvent(type, opts)); } catch {}
    });
    try { target.click?.(); } catch {}
    if (label) log(`点击：${label}；坐标=(${x},${y})；元素=${cssPath(target)}`, 'info');
    return true;
  }

  function clickAtElementCenter(el, label = '') {

      div.className = `dyfe-log-line dyfe-${level}`;

    }
  }

  function makeDraggable(panel, handle) {
    let dragging = false, sx = 0, sy = 0, sl = 0, st = 0;
    const onDown = (e) => {
      if (e.target.tagName === 'BUTTON') return;
      dragging = true;
      const r = panel.getBoundingClientRect();
      sx = e.clientX; sy = e.clientY; sl = r.left; st = r.top;
      panel.style.right = 'auto'; panel.style.bottom = 'auto'; panel.style.left = sl + 'px'; panel.style.top = st + 'px';
      e.preventDefault();
    };
    const onMove = (e) => {
      if (!dragging || APP.destroyed) return;
      panel.style.left = Math.max(0, Math.min(window.innerWidth - 80, sl + e.clientX - sx)) + 'px';
      panel.style.top = Math.max(0, Math.min(window.innerHeight - 34, st + e.clientY - sy)) + 'px';
    };
    const onUp = () => { dragging = false; };
    handle.addEventListener('mousedown', onDown);
    window.addEventListener('mousemove', onMove);
    window.addEventListener('mouseup', onUp);
    APP.cleanups.push(() => handle.removeEventListener('mousedown', onDown));
    APP.cleanups.push(() => window.removeEventListener('mousemove', onMove));
    APP.cleanups.push(() => window.removeEventListener('mouseup', onUp));
  }

  function startStatusTimer() {
    if (APP.statusTimer) clearInterval(APP.statusTimer);
    APP.statusTimer = setInterval(() => {
      if (!APP.destroyed) refreshStatus();
    }, 2200);
  }


  function isFavoritePlaybackState(st = null) {
    st = st || detectState();
    if (st.state !== 'D') return false;
    const url = location.href;
    const modal = modalRoot();
    const modalId = modal?.id || '';
    const modalClass = modal?.className || '';
    const hasFavoriteUrl = /showTab=favorite_collection/.test(url) || /showSubTab=favorite_folder/.test(url) || /favorite_collection/.test(url);
    const hasFavoriteModal = /favorite_collection|favorite_folder/.test(modalId) || /favorite_collection|favorite_folder/.test(String(modalClass));
    const hasFavoriteContext = st.topFavoriteActive && st.folderSubActive && st.collectionNavVisible;
    const hasRightFolder = st.rightFolderTabVisible || !!rightFolderTab();
    return !!(hasRightFolder && (hasFavoriteUrl || hasFavoriteModal || hasFavoriteContext));
  }

  function startEligibilityText(st = null) {
    st = st || detectState();
    if (isFavoritePlaybackState(st)) return '可开始爬取';
    if (st.state === 'D') return '不可爬取：当前播放页不是收藏夹来源';
    return '不可爬取：需先人工进入目标收藏夹并打开视频播放页';
  }

  function renderTextIfChanged(el, text, cacheKey) {
    if (!el) return;
    if (APP[cacheKey] === text) return;
    APP[cacheKey] = text;
    el.textContent = text;
  }

  function runStateText() {
    if (APP.running) return APP.paused ? '已暂停' : '爬取中';
    if (APP.busy) return '执行中';
    if (APP.stopping) return '停止中';
    return '空闲';
  }

  function refreshStatus(st = null, overrideText = '') {
    if (APP.destroyed) return;
    st = st || detectState();
    APP.lastState = st;
    const box = document.getElementById('__dy_fav_export_121_statebox__');
    const stat = document.getElementById('__dy_fav_export_121_stat__');
    const target = getTargetName() || APP.targetName || '-';
    const pageLoadingText = st.loading ? '加载中' : '页面稳定';
    const stateText = overrideText || `界面：${st.label}｜页面：${pageLoadingText}｜${startEligibilityText(st)}｜目标：${target}`;
    const statText = `运行：${runStateText()}｜已采集：${APP.results.size}｜目标：${APP.expectedCount || '-'}`;
    renderTextIfChanged(box, stateText, 'lastStateText');
    renderTextIfChanged(stat, statText, 'lastStatText');
    setRunButtons();
  }

  function parseVideoIdFromUrl(url = location.href) {
    const m = String(url).match(/[?&]modal_id=(\d{8,})|\/video\/(\d{8,})|\/note\/(\d{8,})/);
    return m ? (m[1] || m[2] || m[3]) : '';
  }

  function parseActiveVideoId() {
    const fromUrl = parseVideoIdFromUrl(location.href);
    if (fromUrl) return fromUrl;
    const active = pickVisible('[data-e2e="feed-active-video"][data-e2e-vid], #slideMode[data-e2e-vid], #sliderVideo[data-e2e-vid]');
    const vid = active?.getAttribute('data-e2e-vid');
    if (vid) return vid;
    const clsNode = pickVisible('[class*="video_"]');
    const cls = clsNode?.className || '';
    const m = String(cls).match(/video_(\d{8,})/);
    return m ? m[1] : '';
  }

  function getVisibleTabs() {

  function rectObj(el) {

  }

  function countRealDocLinks() {
    return collectLinkItems(document, { rightOnly: false, includeClickNeeded: false }).length;

  function detectState() {
    const url = location.href;
    const title = document.title || '';
    const hasModalId = /[?&]modal_id=\d+/.test(url);
    const modalVisible = !!modalRoot();
    const activeVideoId = parseActiveVideoId();
    const rightCardVisible = !!rightRoot();
    const rightFolderTabVisible = !!rightFolderTab();
    const rightFolderTabActive = rightFolderActive();
    const topFavActive = topFavoriteActive();
    const folderSubActive = tabSelected(subFolderTab()) || /showSubTab=favorite_folder/.test(url);
    const videoSubActive = tabSelected(subVideoTab()) || /showSubTab=video/.test(url);
    const navVisible = !!collectionNav();
    const leftNavVisible = !!leftNavigation();
    const tabs = getVisibleTabs();
    const bodyText = textOf(document.body).slice(0, 3000);
    const loading = /加载中|正在加载|骨架/.test(bodyText) || !!pickVisible('[class*="loading" i], [id*="loading" i]');
    const realLinks = countRealDocLinks();

    let state = 'O';
    let label = '无法识别';
    let reason = '未匹配可操作状态';

    if ((hasModalId || modalVisible || activeVideoId) && rightCardVisible) {
      state = 'D'; label = '视频详情页-右侧收藏夹栏'; reason = '检测到视频详情/播放层与右侧栏';
    } else if (/\/user\/self/.test(url) && topFavActive && folderSubActive && navVisible && realLinks > 0) {

  function normalizeTypeAndId(raw) {
    const s = String(raw || '');
    let m = s.match(/\/video\/(\d{8,})/);
    if (m) return { type: 'video', id: m[1] };
    m = s.match(/\/note\/(\d{8,})/);
    if (m) return { type: 'note', id: m[1] };
    m = s.match(/[?&]modal_id=(\d{8,})/);
    if (m) return { type: 'video', id: m[1] };
    return null;
  }

  function cleanTitle(s) {
    let t = String(s || '').replace(/\s+/g, ' ').trim();
    t = t.replace(/^(\d+(\.\d+)?万?|播放中|正在播放|已选中|听抖音|点击推荐)\s*/g, '').trim();
    t = t.replace(/\b\d{2}:\d{2}\s*\/\s*\d{2}:\d{2}\b/g, '').replace(/倍速|智能|清屏|连播|顺序连播/g, '').trim();
    if (t.length > 220) t = t.slice(0, 220);
    return t;
  }

  function rejectUrl(url) {
    if (!url) return '空链接';
    if (/source=Baiduspider/i.test(url)) return 'SEO/Baiduspider 链接';
    if (/\/search\//.test(url) || /keyword=/.test(url)) return '搜索链接';
    if (!/\/video\/\d+|\/note\/\d+/.test(url)) return '非作品链接';
    return '';
  }

  function getTitleFromCard(el) {
    if (!el) return '';
    const imgAlt = [...el.querySelectorAll('img[alt]')].map((img) => img.getAttribute('alt')).find((x) => x && x.trim().length > 4);
    return cleanTitle(imgAlt || textOf(el));
  }

  function extractUrlFromElement(el) {
    const a = el?.matches?.('a') ? el : el?.querySelector?.('a[href*="/video/"], a[href*="/note/"]');
    let href = a?.href || a?.getAttribute?.('href') || '';
    if (!href) return '';
    if (href.startsWith('//')) href = location.protocol + href;
    if (href.startsWith('/')) href = location.origin + href;
    return href;
  }

  function collectLinkItems(root = document, opts = {}) {
    const includeClickNeeded = opts.includeClickNeeded !== false;
    const rightOnly = !!opts.rightOnly;
    const cards = new Set();
    const selectors = [
      'a[href*="/video/"]', 'a[href*="/note/"]',
      'a.TyuBARdT', 'a.IdxE71f8', 'li', '[data-e2e="related-user-card-list"] li'
    ];
    for (const sel of selectors) {
      try { root.querySelectorAll(sel).forEach((el) => cards.add(el)); } catch {}
    }
    const out = [];
    for (const el of cards) {
      if (!visible(el) || isInPanel(el)) continue;
      if (rightOnly && !isRightSideElement(el)) continue;
      const r = el.getBoundingClientRect();
      if (r.width < 20 || r.height < 20) continue;
      const url = extractUrlFromElement(el);
      const bad = rejectUrl(url);
      if (url && !bad) {
        const info = normalizeTypeAndId(url);
        if (!info) continue;
        out.push({ type: info.type, id: info.id, link: standardLink(info.type, info.id), title: getTitleFromCard(el), el, source: 'href' });
      } else if (includeClickNeeded && rightOnly) {
        const txt = textOf(el);
        if (/播放中|正在播放/.test(txt)) continue;
        if (r.width >= 65 && r.height >= 65) out.push({ type: 'video', id: '', link: '', title: getTitleFromCard(el), el, source: 'click-needed' });
      }
    }
    const seen = new Set();
    return out.filter((x) => {
      const k = x.link || cssPath(x.el) + '|' + Math.round(x.el.getBoundingClientRect().top);
      if (seen.has(k)) return false;
      seen.add(k);
      return true;
    });
  }

  function isRightSideElement(el) {
    const root = rightRoot();
    if (root && root.contains(el)) return true;
    const r = el.getBoundingClientRect();
    return r.left > window.innerWidth * 0.58;
  }

  function getCurrentVideoTitle() {
    const active = pickVisible('[data-e2e="feed-active-video"]') || pickVisible('#slideMode') || pickVisible('#sliderVideo') || modalRoot();
    if (!active) return '';

    return candidates[0] || '';
  }

  function addResult(item) {
    if (!item?.id) return false;
    const key = `${item.type || 'video'}:${item.id}`;
    const old = APP.results.get(key);
    const value = {
      type: item.type || 'video',
      id: item.id,
      link: item.link || standardLink(item.type || 'video', item.id),
      title: cleanTitle(item.title || ''),
      source: item.source || '',
      time: new Date().toISOString(),
    };
    if (old) {
      if (!old.title && value.title) APP.results.set(key, { ...old, title: value.title });
      return false;
    }
    APP.results.set(key, value);
    refreshStatus();
    return true;
  }

  async function waitFor(predicate, timeout = 10000, interval = 250, label = '') {
    const start = Date.now();
    let last = null;
    while (Date.now() - start < timeout) {
      if (APP.stopping) return null;
      try {
        last = predicate();
        if (last) return last;
      } catch (e) { last = e; }
      await sleep(interval);
    }
    if (label) log(`等待超时：${label}`, 'warn', { last });
    return null;
  }

      await sleep(800);
    }
  }

  function findCollectionElement(name) {
    const nav = collectionNav();
    const roots = [nav, document.body].filter(Boolean);
    const scored = [];
    for (const root of roots) {
      const nodes = [...root.querySelectorAll('a, button, [role="button"], div, p, span')].filter(visible);
      for (const el of nodes) {
        if (isInPanel(el)) continue;
        const t = textOf(el);
        if (!t.includes(name) || /新建收藏夹/.test(t)) continue;
        const r = el.getBoundingClientRect();
        if (r.width < 35 || r.height < 12) continue;
        let score = 0;
        if (root === nav) score += 80;
        if (t === name) score += 50;
        if (new RegExp(name.replace(/[.*+?^${}()|[\]\\]/g, '\\$&') + '\\s*\\d+').test(t)) score += 25;
        if (clickable(el)) score += 10;
        if (r.left > 70 && r.top > 250) score += 8;
        score -= Math.min(t.length / 20, 10);
        scored.push({ el, score, text: t, rect: r });
      }
    }
    scored.sort((a, b) => b.score - a.score);
    const best = scored[0]?.el;
    return best ? findClickableAncestor(best, 7) : null;
  }

      .filter(visible)
      .filter((li) => {
        const t = textOf(li);
        const r = li.getBoundingClientRect();
        if (/播放中|正在播放/.test(t)) return false;
        return r.width > 55 && r.height > 55 && r.left > window.innerWidth * 0.55;
      })
      .sort((a, b) => a.getBoundingClientRect().top - b.getBoundingClientRect().top || a.getBoundingClientRect().left - b.getBoundingClientRect().left);
    return cards[0] || null;
  }

  async function waitVideoIdChanged(beforeId, timeout = 8000, label = '') {
    const changed = await waitFor(() => {
      const id = parseActiveVideoId();
      return id && id !== beforeId ? id : null;
    }, timeout, 250, label || '等待视频切换');
    if (changed) await sleep(900);
    return changed;
  }

  async function switchToNextVideo() {
    const beforeId = parseActiveVideoId();
    if (!beforeId) {
      log('切换前未读取到 activeVideoId，无法切换。', 'error');
      return null;
    }

    log(`切换前ID：${beforeId}；动作：ArrowDown`, 'info');
    dispatchArrowDown();
    let afterId = await waitVideoIdChanged(beforeId, 8500, 'ArrowDown 后 ID 变化');
    if (afterId) {
      log(`切换成功：ArrowDown；${beforeId} → ${afterId}`, 'ok');
      return afterId;
    }
    log(`ArrowDown 未切换：${beforeId} → ${parseActiveVideoId() || '-'}`, 'warn');

    log(`备用动作：鼠标滚轮滚动主视频区域；切换前ID=${beforeId}`, 'info');
    dispatchWheelOnMain();
    afterId = await waitVideoIdChanged(beforeId, 7500, '滚轮后 ID 变化');
    if (afterId) {
      log(`切换成功：Wheel；${beforeId} → ${afterId}`, 'ok');
      return afterId;
    }
    log(`Wheel 未切换：${beforeId} → ${parseActiveVideoId() || '-'}`, 'warn');

    const card = findNextRightCard();
    if (card) {
      log(`备用动作：点击右侧下一个非播放中卡片；切换前ID=${beforeId}；元素=${cssPath(card)}`, 'info');
      clickAtElementCenter(card, '右侧下一卡片');
      afterId = await waitVideoIdChanged(beforeId, 9000, '右侧卡片点击后 ID 变化');
      if (afterId) {
        log(`切换成功：右侧卡片；${beforeId} → ${afterId}`, 'ok');
        return afterId;
      }
      log(`右侧卡片点击后 ID 未变化，判定未切换：${beforeId} → ${parseActiveVideoId() || '-'}`, 'warn');
    } else {
      log('未找到可点击的右侧备用卡片。', 'warn');
    }
    return null;
  }

  async function startCrawlStage() {
    if (APP.running || APP.busy) {
      log('当前已有动作在执行。', 'warn');
      return;
    }
    const st = detectState();
    if (!isFavoritePlaybackState(st)) {
      if (st.state === 'D') {
        log('当前是视频播放页，但不是从收藏夹进入的播放页，不能开始爬取。请先人工进入目标收藏夹，再打开其中任意视频播放页。', 'warn', st);
      } else {
        log(`当前不能开始爬取：${st.label}。请先人工进入目标收藏夹，并打开其中任意视频播放页。`, 'warn', st);
      }
      refreshStatus(st);
      return;
    }
    APP.running = true;
    APP.stopping = false;
    APP.paused = false;
    APP.switchFailCount = 0;
    APP.duplicateCount = 0;
    refreshStatus(st, '界面：收藏夹来源视频详情页｜爬取中');
    log('开始爬取：已确认当前为收藏夹来源播放页；从当前视频开始，按 ArrowDown 切换下一条。', 'info');

    try {
      await ensureRightFavoriteTab();
      await waitRightListLoaded();
      await waitCurrentVideoReady();
      const current = currentVideoItem();
      if (current) {
        const added = addResult(current);
        log(`${added ? '新增当前视频' : '当前视频已存在'}：${current.link}`, added ? 'ok' : 'info');
      }

      const maxLoops = Math.max(20, (APP.expectedCount || 180) + 25);
      for (let i = 0; i < maxLoops && APP.running && !APP.stopping; i++) {
        await waitIfPaused();
        if (APP.stopping || !APP.running) break;

        if (APP.expectedCount && APP.results.size >= APP.expectedCount) {
          log(`已达到目标数量：${APP.results.size}/${APP.expectedCount}，停止爬取。`, 'ok');
          break;
        }

        await waitCurrentVideoReady();
        const before = parseActiveVideoId();
        const after = await switchToNextVideo();
        if (!after || after === before) {
          APP.switchFailCount += 1;
          log(`本次未切换成功，连续失败=${APP.switchFailCount}`, 'warn');
          if (APP.switchFailCount >= 4) {
            log('连续多次无法切换，停止爬取。', 'warn');
            break;
          }
          continue;
        }
        APP.switchFailCount = 0;
        await waitCurrentVideoReady();
        const item = currentVideoItem();
        if (!item) {
          log('切换后未读取到当前视频信息。', 'warn');
          continue;
        }
        const added = addResult(item);
        if (added) {
          APP.duplicateCount = 0;
          log(`新增：${item.link}｜标题=${item.title || '未读取到标题'}`, 'ok');
        } else {
          APP.duplicateCount += 1;
          log(`已存在：${item.link}｜重复计数=${APP.duplicateCount}`, 'info');
          if (!APP.expectedCount && APP.duplicateCount >= 8) {
            log('连续读取到重复视频，停止爬取。', 'warn');
            break;
          }
        }
        refreshStatus();
      }
      log(`爬取结束：共缓存 ${APP.results.size} 条。请点击【导出CSV】导出。`, 'ok');
    } catch (e) {
      log(`爬取失败：${e.message || e}`, 'error');
    } finally {
      APP.running = false;
      APP.paused = false;
      APP.stopping = false;
      refreshStatus();
    }
  }

  function togglePause() {
    if (!APP.running) {
      log('当前没有正在爬取的任务，无法暂停/继续。', 'warn');
      return;
    }
    APP.paused = !APP.paused;
    log(APP.paused ? '已暂停。缓存和当前页面保留。' : '已继续。', APP.paused ? 'warn' : 'ok');
    refreshStatus();
  }

  function stopCurrentAction() {
    if (!APP.running && !APP.busy) {
      log('当前没有正在执行的动作。', 'warn');
      return;
    }
    APP.stopping = true;
    APP.paused = false;
    log('已请求停止。当前步骤结束后停止。', 'warn');
    refreshStatus();
  }

  function clearCacheOnly() {
    APP.results.clear();
    APP.expectedCount = 0;
    APP.switchFailCount = 0;
    APP.duplicateCount = 0;
    APP.lockedRightScrollerSelector = '';
    log('已清除缓存：结果、计数和运行状态已清空；日志保留。', 'warn');
    refreshStatus();
  }

  function makeRows() {
    return [...APP.results.values()].map((x, idx) => ({
      序号: idx + 1,
      '标题/文案预览': cleanTitle(x.title || ''),
      类型: x.type || 'video',
      链接: x.link,
    }));
  }

  function csvEscape(v) {
    const s = String(v ?? '');
    return /[",\n\r]/.test(s) ? '"' + s.replace(/"/g, '""') + '"' : s;
  }

  function exportCsv(manual = true) {
    if (!APP.results.size) {
      log('没有可导出的缓存结果。', 'warn');
      return;
    }
    const rows = makeRows();
    const header = ['序号', '标题/文案预览', '类型', '链接'];
    const csv = '\ufeff' + [header.join(','), ...rows.map((r) => header.map((h) => csvEscape(r[h])).join(','))].join('\r\n');
    const blob = new Blob([csv], { type: 'text/csv;charset=utf-8;' });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    const name = APP.targetName || getTargetName() || '收藏夹';
    const ts = new Date().toISOString().slice(0, 16).replace(/[-T:]/g, '').slice(0, 12);
    a.href = url;
    a.download = `抖音收藏夹_${name}_${ts}.csv`;
    document.body.appendChild(a);
    a.click();
    a.remove();
    setTimeout(() => URL.revokeObjectURL(url), 1000);
    log(`已导出 CSV：${rows.length} 条。`, manual ? 'ok' : 'info');
  }

  async function copyText(text) {
    try {
      await navigator.clipboard.writeText(text);
      return true;
    } catch {
      const ta = document.createElement('textarea');
      ta.value = text;
      document.body.appendChild(ta);
      ta.select();
      const ok = document.execCommand('copy');
      ta.remove();
      return ok;
    }
  }

  async function copyLinks() {
    const links = [...APP.results.values()].map((x) => x.link).join('\n');
    if (!links) {
      log('没有可复制的链接。', 'warn');
      return;
    }
    const ok = await copyText(links);
    log(ok ? `已复制 ${APP.results.size} 条链接。` : '复制失败。', ok ? 'ok' : 'error');
  }

  function boot() {
    createPanel();
  }

  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', boot);
  } else {
    boot();
  }
})();
