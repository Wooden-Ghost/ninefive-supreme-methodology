  async function waitFor(predicate, timeout = PAGE_STABLE_TIMEOUT, interval = POLL_INTERVAL) {
    const started = Date.now();
    let lastError;
    while (Date.now() - started < timeout) {
      try {
        const value = await predicate();

  function normalize(value) {
    return String(value == null ? '' : value).replace(/\s+/g, ' ').trim();
  }

  function normalizeHeader(value) {
    return normalize(value).toLowerCase().replace(/[\s_\-｜|:：]/g, '');
  }

  function dashToEmpty(value) {
    const text = normalize(value);
    return text === '-' ? '' : text;
  }

  function textOf(el) {
    return normalize(el ? (el.innerText || el.textContent || '') : '');
  }

  function stripRequiredStar(label) {
    return normalize(label).replace(/\s*\*\s*$/, '');
  }

  function stripQuotes(value) {
    const text = normalize(value);
    if ((text.startsWith('"') && text.endsWith('"')) || (text.startsWith("'") && text.endsWith("'"))) {
      return text.slice(1, -1).trim();
    }
    return text;
  }

  function baseCompanyName(value) {
    return normalize(value)
      .replace(/（[^）]*）/g, '')
      .replace(/\([^)]*\)/g, '')
      .replace(/[.，,]+$/g, '')
      .replace(/\s+/g, ' ')
      .trim()
      .toUpperCase();
  }

  function simplifiedCompanyName(value) {
    const text = baseCompanyName(value)
      .replace(/[\u3000]/g, ' ')
      .replace(/[.,，。'’`&/\\-]+/g, ' ')
      .replace(/\s+/g, ' ')
      .trim();
    const suffixes = new Set([
      'INC', 'INCORPORATED', 'LLC', 'LTD', 'LIMITED', 'CO', 'COMPANY',
      'CORP', 'CORPORATION', 'GROUP', 'SAS', 'SA', 'CV'
    ]);
    const parts = text.split(' ').filter(Boolean);
    while (parts.length > 1 && suffixes.has(parts[parts.length - 1])) {
      parts.pop();
    }
    return parts.join(' ');
  }

  function namesAreSimilar(a, b) {
    const baseA = baseCompanyName(a);
    const baseB = baseCompanyName(b);
    if (!baseA || !baseB) return false;
    if (baseA === baseB) return true;
    const simpleA = simplifiedCompanyName(a);
    const simpleB = simplifiedCompanyName(b);
    if (simpleA && simpleA === simpleB) return true;
    const minLen = Math.min(baseA.length, baseB.length);
    return minLen >= 8 && (baseA.includes(baseB) || baseB.includes(baseA));
  }

  function isVisible(el) {
    if (!el) return false;
    const rect = el.getBoundingClientRect();
    const style = window.getComputedStyle(el);
    return rect.width > 0 && rect.height > 0 && style.visibility !== 'hidden' && style.display !== 'none';
  }

  function setNativeValue(el, value) {
    if (!el) throw new Error('目标输入框不存在');
    const proto = el instanceof HTMLTextAreaElement ? HTMLTextAreaElement.prototype : HTMLInputElement.prototype;
    const setter = Object.getOwnPropertyDescriptor(proto, 'value')?.set;
    if (setter) setter.call(el, value);
    else el.value = value;
    try {
      el.dispatchEvent(new InputEvent('input', { bubbles: true, composed: true, inputType: 'insertText', data: String(value || '') }));
    } catch (_) {
      el.dispatchEvent(new Event('input', { bubbles: true, composed: true }));
    }
    el.dispatchEvent(new Event('change', { bubbles: true, composed: true }));
  }

  function pressEnter(el) {
    ['keydown', 'keypress', 'keyup'].forEach(type => {
      el.dispatchEvent(new KeyboardEvent(type, {
        key: 'Enter',
        code: 'Enter',
        keyCode: 13,
        which: 13,
        bubbles: true,
        cancelable: true,
        composed: true
      }));
    });
  }

  function pressCtrlA(el) {
    ['keydown', 'keyup'].forEach(type => {
      el.dispatchEvent(new KeyboardEvent(type, {
        key: 'a',
        code: 'KeyA',
        keyCode: 65,
        which: 65,
        ctrlKey: true,
        bubbles: true,
        cancelable: true
      }));
    });
  }

  function pressBackspace(el) {
    ['keydown', 'keyup'].forEach(type => {
      el.dispatchEvent(new KeyboardEvent(type, {
        key: 'Backspace',
        code: 'Backspace',
        keyCode: 8,
        which: 8,
        bubbles: true,
        cancelable: true
      }));
    });
  }

  function clickLikeUser(el) {
    if (!el) throw new Error('点击目标不存在');
    el.scrollIntoView({ block: 'center', inline: 'center' });
    const realWin = el.ownerDocument?.defaultView || window;
    const rect = el.getBoundingClientRect();
    const clientX = rect.left + Math.max(4, Math.min(rect.width - 4, rect.width / 2));
    const clientY = rect.top + Math.max(4, Math.min(rect.height - 4, rect.height / 2));
    ['pointerdown', 'mousedown', 'mouseup', 'click'].forEach(type => {
      try {
        el.dispatchEvent(new realWin.MouseEvent(type, {
          bubbles: true,
          cancelable: true,
          view: realWin,
          clientX,
          clientY
        }));
      } catch (_) {
        el.dispatchEvent(new MouseEvent(type, { bubbles: true, cancelable: true, clientX, clientY }));
      }
    });
    if (typeof el.click === 'function') {
      try { el.click(); } catch (_) {}
    }
  }

  function findSearchInput() {
    return document.querySelector(CRM_PATHS.searchInput);
  }

  function findCountText() {
    const body = document.body?.innerText || '';
    return (body.match(/共\s*\d+\s*条/) || [''])[0];
  }

  function findNoDataText() {
    const preferred = Array.from(document.querySelectorAll('.n-data-table-empty, .empty, .no-data'))
      .find(el => isVisible(el) && /暂无数据|无数据|No Data/i.test(textOf(el)));
    if (preferred) return preferred;

    return Array.from(document.querySelectorAll('span, div, td, p'))
      .find(el => {
        if (!isVisible(el)) return false;
        const text = textOf(el);
        if (!/^(暂无数据|无数据|No Data)$/i.test(text)) return false;
        const rect = el.getBoundingClientRect();
        return rect.width > 20 && rect.height > 10;
      }) || null;
  }

  function getVisibleExactText(text) {
    return Array.from(document.querySelectorAll('button, a, span, div'))
      .filter(el => isVisible(el) && textOf(el) === text);
  }

  function hasActiveFilterHint() {
    return getVisibleExactText('清空筛选').length > 0;
  }

  function readTableState() {
    const headerCells = Array.from(document.querySelectorAll(CRM_PATHS.headerTh));
    const headers = headerCells.map(th => ({
      text: textOf(th),
      colKey: th.getAttribute('data-col-key') || ''
    }));

    const rows = Array.from(document.querySelectorAll(CRM_PATHS.visibleRows))
      .map((tr, rowIndex) => {
        const nameCell = tr.querySelector(CRM_PATHS.rowNameCell);
        const industryCell = tr.querySelector(CRM_PATHS.rowIndustryCell);
        const priorityCell = tr.querySelector(CRM_PATHS.rowPriorityCell);
        const operationCell = tr.querySelector(CRM_PATHS.rowOperationCell);
        if (!nameCell) return null;

        const name = textOf(nameCell);
        const industry = textOf(industryCell);
        const priority = textOf(priorityCell);
        const obj = {
          '客户名称': name,
          '行业': industry,
          '客户优先级': priority
        };

          rowIndex,
          tr,
          nameCell,
          industryCell,
          priorityCell,
          operationCell,
          editButton: tr.querySelector(CRM_PATHS.rowEditButton),
          obj,
          vals: [name, industry, priority],

      headers: headers.map(item => item.text).filter(Boolean),
      headerMap: headers,

      searchValue: findSearchInput()?.value || ''

  }

    return JSON.stringify({

  function looksLoading() {
    const text = document.body?.innerText || '';
    if (/加载中|Loading/i.test(text)) return true;
    const selectors = [
      '.n-spin--show',
      '.n-data-table-loading',
      '.n-base-loading',
      '[aria-busy="true"]'
    ];
    return selectors.some(sel => Array.from(document.querySelectorAll(sel)).some(el => {
      if (!isVisible(el)) return false;
      const style = window.getComputedStyle(el);
      return style.display !== 'none' && style.visibility !== 'hidden';
    }));
  }

  function hasTargetInVisibleRows(targetName) {
    if (!targetName) return false;
    const targetExact = normalize(targetName).toUpperCase();
    const targetBase = baseCompanyName(targetName);
    return readTableState().rows.some(row => {
      const name = row.obj['客户名称'];
      return normalize(name).toUpperCase() === targetExact || baseCompanyName(name) === targetBase;
    });
  }

  function forceFocusSearchInput(input) {
    input.scrollIntoView({ block: 'center', inline: 'center' });
    input.focus();
    input.click();
    try { input.select?.(); } catch (_) {}
  }

  async function triggerSearch(input, label = '') {
    // 搜索触发只使用已验证可用的输入框 Enter，不猜测搜索图标路径。
    await sleep(120);
    pressEnter(input);
    input.dispatchEvent(new Event('input', { bubbles: true, composed: true }));
    input.dispatchEvent(new Event('change', { bubbles: true, composed: true }));
    input.dispatchEvent(new Event('blur', { bubbles: true, composed: true }));
    await sleep(label ? 900 : 500);
  }

  function searchApplied(previousContentSignature, targetName, options = {}) {
    const stateNow = readTableState();
    const contentSig = tableContentSignature();
    const noData = !!findNoDataText() || /共\s*0\s*条/.test(stateNow.countText || '');
    const contentChanged = previousContentSignature && contentSig !== previousContentSignature;
    const inputMatched = normalize(stateNow.searchValue).toUpperCase() === normalize(targetName).toUpperCase();
    const targetVisible = hasTargetInVisibleRows(targetName);
    const allowUnchangedTarget = !!options.allowUnchangedTarget;

    if (!targetName) return contentChanged || (!previousContentSignature && (stateNow.rows.length > 0 || noData));

    // 初次搜索：优先要求表格内容发生变化后出现目标行/暂无数据，避免拿全量旧列表误判。
    if (inputMatched && contentChanged && (targetVisible || noData)) return true;

    // 保存后回读或当前列表已经停在同一个搜索结果时，允许“未变化但目标已可见”。
    if (inputMatched && allowUnchangedTarget && (targetVisible || noData)) return true;

    // 部分 CRM 场景下搜索结果只有一条且输入框已是目标名，即使虚拟表格 signature 未变，也可视为目标结果可读。
    if (inputMatched && targetVisible && stateNow.rows.length === 1 && /共\s*1\s*条/.test(stateNow.countText || '')) return true;

    return false;
  }

  async function waitForPageStable(label = '页面', timeout = PAGE_STABLE_TIMEOUT) {
    const started = Date.now();
    let lastSignature = '';
    let stableSince = 0;
    while (Date.now() - started < timeout) {
      const sig = tableSignature() + '|' + (!!findEditForm()) + '|' + (document.body?.innerText || '').length;
      const loading = looksLoading();
      if (!loading && sig === lastSignature) {
        if (!stableSince) stableSince = Date.now();
        if (Date.now() - stableSince >= STABLE_HOLD_MS) return true;
      } else {
        lastSignature = sig;
        stableSince = 0;
      }
      await sleep(POLL_INTERVAL);
    }
    throw new Error(`${label}等待稳定超时（${Math.round(timeout / 1000)}s）`);
  }

  async function waitForSearchResultStable(label = '搜索结果', previousContentSignature = '', targetName = '', options = {}) {
    const started = Date.now();
    let lastSignature = '';
    let stableSince = 0;
    let applied = false;

    while (Date.now() - started < PAGE_STABLE_TIMEOUT) {
      const sig = tableSignature();
      const loading = looksLoading();

      if (searchApplied(previousContentSignature, targetName, options)) applied = true;

      if (sig === lastSignature) {
        if (!stableSince) stableSince = Date.now();
        const stableEnough = Date.now() - stableSince >= STABLE_HOLD_MS;
        const loadingCanIgnore = !loading || Date.now() - started >= 1800;
        if (applied && stableEnough && loadingCanIgnore) return true;
      } else {
        lastSignature = sig;
        stableSince = 0;
      }
      await sleep(POLL_INTERVAL);
    }

    const stateNow = readTableState();
    if (searchApplied(previousContentSignature, targetName, options)) return true;

    const targetInfo = targetName ? `：${targetName}` : '';
    const countInfo = stateNow.countText || `${stateNow.rows.length} 条`;
    throw new Error(`${label}未得到目标结果${targetInfo}；当前结果（${countInfo}），已停止读取，避免拿旧列表误判`);
  }

  async function setSearchValueAndRun(input, value, label) {
    forceFocusSearchInput(input);
    setNativeValue(input, value);
    await triggerSearch(input, label);

    if (normalize(input.value) !== normalize(value)) {
      forceFocusSearchInput(input);
      setNativeValue(input, value);
      await triggerSearch(input, label);
    }
  }

  async function clearSearch() {
    const input = findSearchInput();
    if (!input) throw new Error(`找不到“通过名称搜索”搜索框：${CRM_PATHS.searchInput}`);
    forceFocusSearchInput(input);

    setNativeValue(input, '');
    await triggerSearch(input, '清空搜索');
    await waitFor(() => normalize(input.value) === '', 2500, 100).catch(() => null);
  }

  async function prepareSearchBase(input) {
    forceFocusSearchInput(input);
    const beforeValue = normalize(input.value);
    if (!beforeValue) {
      return;
    }

    try {
      pressCtrlA(input);
      await sleep(60);
      pressBackspace(input);
      await sleep(80);
    } catch (_) {}
    setNativeValue(input, '');
    await triggerSearch(input, '清空搜索');
    await waitFor(() => normalize(input.value) === '', 2500, 100).catch(() => null);
  }

  async function searchCompany(name, options = {}) {
    const input = findSearchInput();
    if (!input) throw new Error(`找不到“通过名称搜索”搜索框：${CRM_PATHS.searchInput}`);

    const currentBefore = readTableState();
    const alreadyOnTarget = normalize(currentBefore.searchValue).toUpperCase() === normalize(name).toUpperCase()
      && hasTargetInVisibleRows(name);

    // 回读验证场景：当前列表已经是目标搜索结果时，不重复搜索。
    if (options.allowCurrentResult && alreadyOnTarget) {
      await waitForSearchResultStable('当前列表结果', tableContentSignature(), name, { allowUnchangedTarget: true });
      return readTableState();
    }

    await prepareSearchBase(input);
    const baseSignature = tableContentSignature();
    await setSearchValueAndRun(input, name, name);

    try {
      await waitForSearchResultStable('搜索结果', baseSignature, name, { allowUnchangedTarget: !!options.allowUnchangedTarget });
    } catch (err) {
      log(`搜索未得到目标结果，二次触发：${name}`, 'warn');
      await setSearchValueAndRun(input, name, name);
      await waitForSearchResultStable('二次搜索结果', baseSignature, name, { allowUnchangedTarget: !!options.allowUnchangedTarget });
    }

    return readTableState();
  }

  async function verifyCurrentOrSearch(targetName) {
    await waitForListReadyAfterSave(targetName);

    let currentState = readTableState();
    let currentRows = findAcceptableRows(currentState.rows, targetName, true);
    if (currentRows.length === 1) {
      log('当前列表已包含目标行，直接回读验证。', 'info');
      return { tableState: currentState, rows: currentRows, source: 'current_list' };
    }

    log(`当前列表未直接读到唯一目标，重新搜索回读：${targetName}`, 'warn');
    currentState = await searchCompany(targetName, { allowCurrentResult: true, allowUnchangedTarget: true });
    currentRows = findAcceptableRows(currentState.rows, targetName, true);
    return { tableState: currentState, rows: currentRows, source: 'search' };
  }

  function findAcceptableRows(rows, targetName, allowBracketMatch = true) {
    const targetExact = normalize(targetName).toUpperCase();
    const targetBase = baseCompanyName(targetName);

    const exact = rows.filter(row => normalize(row.obj['客户名称']).toUpperCase() === targetExact);
    if (exact.length) return exact.map(row => ({ ...row, matchMode: 'exact' }));

    if (allowBracketMatch) {
      const base = rows.filter(row => baseCompanyName(row.obj['客户名称']) === targetBase);
      if (base.length) return base.map(row => ({ ...row, matchMode: 'base_equal' }));
    }

    return [];
  }

  function shouldReportAmbiguity(tableState, acceptedRow, targetName) {
    if (!acceptedRow || tableState.rows.length <= 1) return false;
    if (acceptedRow.matchMode !== 'exact') return true;
    const acceptedName = acceptedRow.obj['客户名称'];
    const similar = tableState.rows.filter(row => {
      const name = row.obj['客户名称'];
      if (normalize(name).toUpperCase() === normalize(acceptedName).toUpperCase()) return false;
      return namesAreSimilar(name, acceptedName) || namesAreSimilar(name, targetName);
    });
    return similar.length > 0;
  }

  function splitInputLine(line) {
    if (line.includes('\t')) return line.split('\t');
    if (line.includes('|')) return line.split('|');
    return line.split(',');
  }

  function detectHeaderIndexes(parts) {
    const headers = parts.map(normalizeHeader);
    const findAny = aliases => headers.findIndex(h => aliases.includes(h));
    const nameIndex = findAny(['客户名称', '公司名称', '名字', '名称']);
    const industryIndex = findAny(['行业']);
    const priorityIndex = findAny(['rating', '客户优先级', '优先级', '评级']);
    if (nameIndex >= 0 && industryIndex >= 0 && priorityIndex >= 0) {
      return { nameIndex, industryIndex, priorityIndex, source: 'header' };
    }
    return null;
  }

  function inferIndexes(parts) {
    if (parts.length >= 7) {
      return { nameIndex: 0, industryIndex: 3, priorityIndex: 6, source: 'wps_B_to_H' };
    }
    if (parts.length >= 3) {
      return { nameIndex: 0, industryIndex: 1, priorityIndex: 2, source: 'three_columns' };
    }
    return null;
  }

  function validateIndustry(value, lineNumber) {
    const industry = normalize(value);
    if (!industry) throw new Error(`第 ${lineNumber} 行行业为空`);
    if (!INDUSTRY_ALLOW_SET.has(industry.toLowerCase())) {
      throw new Error(`第 ${lineNumber} 行行业不在允许范围内：${industry}`);
    }
    return industry;
  }

  function validatePriority(value, lineNumber) {
    const priority = normalize(value).toUpperCase();
    if (!priority) throw new Error(`第 ${lineNumber} 行 rating/客户优先级为空`);
    if (!/^[A-D]$/.test(priority)) {
      throw new Error(`第 ${lineNumber} 行 rating/客户优先级不是 A/B/C/D：${value}`);
    }
    return priority;
  }

  function parseInput() {
    const text = panelEls.input.value || '';
    const lines = text.split(/\r?\n/);
    const targets = [];
    const errors = [];
    let indexes = null;
    let indexSource = '';

    lines.forEach((line, idx) => {
      const lineNumber = idx + 1;
      const raw = line.trim();
      if (!raw || raw.startsWith('#')) return;

      const parts = splitInputLine(line).map(stripQuotes);
      const headerIndexes = detectHeaderIndexes(parts);
      if (headerIndexes) {
        indexes = headerIndexes;
        indexSource = headerIndexes.source;
        return;
      }

      const useIndexes = indexes || inferIndexes(parts);
      if (!useIndexes) {
        errors.push(`第 ${lineNumber} 行列数不足，无法识别客户名称/行业/rating`);
        return;
      }
      if (!indexes) {
        indexes = useIndexes;
        indexSource = useIndexes.source;
      }

      try {
        const name = stripQuotes(parts[useIndexes.nameIndex]);
        const industry = validateIndustry(parts[useIndexes.industryIndex], lineNumber);
        const priority = validatePriority(parts[useIndexes.priorityIndex], lineNumber);
        if (!name) throw new Error(`第 ${lineNumber} 行客户名称为空`);
        targets.push({ lineNumber, name, industry, priority });
      } catch (err) {
        errors.push(err.message || String(err));
      }
    });

    return { targets, errors, indexSource };
  }

  function formatVisibleRows(rows) {
    return rows.slice(0, 8).map(row => {
      const name = row.obj['客户名称'] || '';
      const industry = row.obj['行业'] || '';
      const priority = row.obj['客户优先级'] || '';
      return `${name} [行业:${industry || '-'} 优先级:${priority || '-'}]`;
    }).join(' / ');
  }

  function summarizeResult(result) {
    const changed = result.changed?.map(item => `${item.field}:${item.from || '-'} -> ${item.to}`).join('; ') || '';
    const note = result.note ? `；${result.note}` : '';
    if (result.status === 'updated_verified') return `已更新并验证 ${changed}${note}`;
    if (result.status === 'already_match_verified') return `已一致，未写入${note}`;
    if (result.status === 'skipped_ambiguous') return `跳过：无法唯一确认；${result.note || ''}`;
    if (result.status === 'verify_failed') return `保存后回读不一致；${result.note || ''}`;
    if (result.status === 'error') return `异常：${result.error}`;
    return `${result.status || '未知状态'}${note}`;
  }

  async function processTarget(target) {
    const result = {
      index: target.lineNumber,
      name: target.name,
      industry: target.industry,
      priority: target.priority,
      status: '',
      changed: [],
      note: ''
    };

    const tableState = await searchCompany(target.name);
    const acceptedRows = findAcceptableRows(tableState.rows, target.name, true);

    result.searchCountText = tableState.countText;
    result.searchRows = tableState.rows.map(row => row.obj['客户名称']).slice(0, 12);

    if (acceptedRows.length !== 1) {
      result.status = 'skipped_ambiguous';
      const filterNote = tableState.rows.length === 0 && hasActiveFilterHint()
        ? '；当前检测到“清空筛选”，筛选可能导致搜不到客户'
        : '';
      result.note = `搜索结果 ${tableState.countText || tableState.rows.length + ' 条'}；可接受匹配 ${acceptedRows.length} 条；可见结果：${formatVisibleRows(tableState.rows) || '无'}${filterNote}`;
      return result;
    }

    const accepted = acceptedRows[0];
    result.crmName = accepted.obj['客户名称'];
    result.matchMode = accepted.matchMode;

    const beforeIndustry = dashToEmpty(accepted.obj['行业']);
    const beforePriority = dashToEmpty(accepted.obj['客户优先级']);
    result.beforeIndustry = beforeIndustry;
    result.beforePriority = beforePriority;

    const needsReport = shouldReportAmbiguity(tableState, accepted, target.name);
    if (needsReport) {
      result.note = `多结果但已按 ${accepted.matchMode === 'exact' ? '完全同名' : '去括号同名'} 处理；可见结果：${formatVisibleRows(tableState.rows)}`;
    }

    if (beforeIndustry === normalize(target.industry) && beforePriority === normalize(target.priority)) {
      result.status = 'already_match_verified';
      result.afterIndustry = beforeIndustry;
      result.afterPriority = beforePriority;
      return result;
    }

    await clickEditInRow(accepted);

    const formState = readFormFields();
    const formName = getFormField(formState.fields, '客户名称');
    result.formName = formName;
    if (baseCompanyName(formName) !== baseCompanyName(target.name)) {
      await cancelFormIfPresent();
      result.status = 'skipped_ambiguous';
      result.note = `编辑页客户名称不匹配：输入=${target.name}，表单=${formName}`;
      return result;
    }

    const currentIndustry = getFormField(formState.fields, '行业');
    const currentPriority = getFormField(formState.fields, '客户优先级');

    if (normalize(currentIndustry) !== normalize(target.industry)) {
      fillFormField('行业', target.industry);
      result.changed.push({ field: '行业', from: currentIndustry, to: target.industry });
      log(`已填写行业：${currentIndustry || '-'} -> ${target.industry}`, 'info');
    }
    if (normalize(currentPriority) !== normalize(target.priority)) {
      fillFormField('客户优先级', target.priority);
      result.changed.push({ field: '客户优先级', from: currentPriority, to: target.priority });
      log(`已填写客户优先级：${currentPriority || '-'} -> ${target.priority}`, 'info');
    }

    if (!result.changed.length) {
      await cancelFormIfPresent();
      result.status = 'already_match_verified';
      result.afterIndustry = currentIndustry;
      result.afterPriority = currentPriority;
      return result;
    }

    log('已填写完成，点击更新。', 'info');
    const saveInfo = await saveForm();
    result.saveInfo = saveInfo;
    log(`已点击更新，保存返回状态：${saveInfo.successText ? '检测到更新成功' : '表单已关闭/返回列表'}`, saveInfo.successText ? 'ok' : 'info');

    const verifyInfo = await verifyCurrentOrSearch(target.name);
    const verifyRows = verifyInfo.rows;
    if (verifyRows.length !== 1) {
      result.status = 'verify_failed';
      result.note = `疑似已保存，但回读未确认；来源=${verifyInfo.source}；结果：${formatVisibleRows(verifyInfo.tableState.rows)}`;
      return result;
    }

    let finalRows = verifyRows;
    const settled = await waitFor(() => {
      const st = readTableState();
      const rows = findAcceptableRows(st.rows, target.name, true);
      if (rows.length !== 1) return null;
      const rowObj = rows[0].obj;
      const industry = dashToEmpty(rowObj['行业']);
      const priority = dashToEmpty(rowObj['客户优先级']);
      if (industry === normalize(target.industry) && priority === normalize(target.priority)) {
        return rows;
      }
      return null;
    }, 4500, 300).catch(() => null);

    if (settled) finalRows = settled;

    const verified = finalRows[0].obj;
    const afterIndustry = dashToEmpty(verified['行业']);
    const afterPriority = dashToEmpty(verified['客户优先级']);
    result.afterIndustry = afterIndustry;
    result.afterPriority = afterPriority;

    if (afterIndustry === normalize(target.industry) && afterPriority === normalize(target.priority)) {
      result.status = 'updated_verified';
      log(`当前列表回读验证通过：行业=${afterIndustry || '-'}，客户优先级=${afterPriority || '-'}`, 'ok');
    } else {
      result.status = 'verify_failed';
      result.note = `疑似已保存，但回读不一致；目标 行业=${target.industry} 优先级=${target.priority}；回读 行业=${afterIndustry || '-'} 优先级=${afterPriority || '-'}`;
    }

    return result;
  }

  function log(message, level = 'info') {
    const time = new Date().toLocaleTimeString();
    const item = { time, level, message };
    state.logs.push(item);
    if (level === 'warn' && state.activeCompanyName) {
      state.currentCompanyWarned = true;
      if (state.inRetryPass) state.retryWarnCompanyNames.add(state.activeCompanyName);
      else state.warnCompanyNames.add(state.activeCompanyName);
    }
    if (!panelEls.log) return;
    const div = document.createElement('div');
    div.className = `crm-batch-log-row ${level}`;
    div.textContent = `[${time}] ${message}`;
    panelEls.log.appendChild(div);
    panelEls.log.scrollTop = panelEls.log.scrollHeight;
  }

  function resetRunLogs() {
    state.logs = [];
    state.results = [];
    state.warnCompanyNames = new Set();
    state.retryWarnCompanyNames = new Set();
    state.retryStillWarnCompanyNames = new Set();
    state.activeCompanyName = '';
    state.currentCompanyWarned = false;
    state.inRetryPass = false;
    state.retryProcessed = 0;
    state.waitingManual = false;
    state.manualContinueResolve = null;
    state.total = 0;
    state.current = 0;
    if (panelEls.log) panelEls.log.innerHTML = '';
    refreshSummary();
  }


  function logWarnCompanySummary(title = '本次黄色日志涉及公司（需人工复核）：', sourceSet = state.warnCompanyNames) {
    const names = Array.from(sourceSet || []);
    if (!names.length) {
      log('本次黄色日志涉及公司：无', 'info');
      return;
    }
    log(title, 'warn');
    names.forEach(name => log(name, 'warn'));
  }

  function logRetryFinalSummary() {
    const names = Array.from(state.retryStillWarnCompanyNames || []);
    if (!names.length) {
      log('复跑后仍为黄色的公司：无', 'info');
      return;
    }
    log('复跑后仍为黄色，已交由人工处理/复核的公司：', 'warn');
    names.forEach(name => log(name, 'warn'));
  }

  async function waitForManualNext(companyName) {
    state.waitingManual = true;
    state.pauseRequested = false;
    setMode('等待人工', 'paused');
    refreshButtons();
    log(`${companyName}：复跑后仍为黄色。请人工处理，完成后点击“下一个”继续。`, 'warn');

    await new Promise(resolve => {
      state.manualContinueResolve = resolve;
    });

    state.manualContinueResolve = null;
    state.waitingManual = false;
    if (state.running && !state.stopRequested) setMode('运行中', 'running');

  }

  function continueAfterManual() {
    if (!state.waitingManual) return;
    log('已点击“下一个”，继续处理后续黄色公司。', 'info');
    const resolve = state.manualContinueResolve;
    if (resolve) resolve();
  }

  async function retryWarnTargets(targets) {
    const warnNames = Array.from(state.warnCompanyNames || []);
    if (!warnNames.length) {
      log('首轮无黄色公司，无需复跑。', 'info');
      return;
    }

    const warnNameSet = new Set(warnNames);
    const retryTargets = targets.filter(target => warnNameSet.has(target.name));
    if (!retryTargets.length) {
      log('首轮存在黄色日志，但未匹配到可复跑的输入行。', 'warn');
      return;
    }

    logWarnCompanySummary('首轮黄色日志涉及公司，开始自动复跑：', warnNameSet);
    state.inRetryPass = true;
    state.retryWarnCompanyNames = new Set();
    state.retryStillWarnCompanyNames = new Set();
    state.retryProcessed = 0;

    for (const target of retryTargets) {
      if (state.stopRequested) break;
      await waitIfPaused();
      if (state.stopRequested) break;

      state.retryProcessed += 1;
      state.activeCompanyName = target.name;
      state.currentCompanyWarned = false;
      refreshSummary();

      let retryHardFailed = false;
      try {
        log(`复跑 ${state.retryProcessed}/${retryTargets.length}：${target.name} -> 行业=${target.industry} 优先级=${target.priority}`, 'info');
        const result = await processTarget(target);
        result.retry = true;
        state.results.push(result);
        const level = ['updated_verified', 'already_match_verified'].includes(result.status)
          ? (result.note ? 'warn' : 'ok')
          : (result.status === 'verify_failed' || result.status === 'error' ? 'error' : 'warn');
        if (result.status === 'verify_failed' || result.status === 'error') retryHardFailed = true;
        log(`${target.name}：复跑结果：${summarizeResult(result)}`, level);
      } catch (err) {
        await cancelFormIfPresent().catch(() => {});
        const result = {
          index: target.lineNumber,
          name: target.name,
          industry: target.industry,
          priority: target.priority,
          status: 'error',
          retry: true,
          error: err && err.stack ? err.stack : String(err)
        };
        retryHardFailed = true;
        state.results.push(result);
        log(`${target.name}：复跑异常 ${err.message || err}`, 'error');
      }

      const stillWarn = state.currentCompanyWarned || state.retryWarnCompanyNames.has(target.name);
      if (stillWarn || retryHardFailed) {
        state.retryStillWarnCompanyNames.add(target.name);
        await waitForManualNext(target.name);
      } else {
        log(`${target.name}：复跑未再出现黄色日志。`, 'ok');
      }

      state.activeCompanyName = '';
      state.currentCompanyWarned = false;
      refreshSummary();
      await sleep(120);
    }

    state.inRetryPass = false;
    state.activeCompanyName = '';
    state.currentCompanyWarned = false;
    logRetryFinalSummary();
  }

  function refreshSummary() {
    const counts = state.results.reduce((acc, item) => {
      acc[item.status] = (acc[item.status] || 0) + 1;
      return acc;
    }, {});
    const processed = state.results.length;
    const text = [
      `总 ${state.total || 0}`,
      `已处理 ${processed}`,
      `更新 ${counts.updated_verified || 0}`,
      `一致 ${counts.already_match_verified || 0}`,
      `跳过 ${counts.skipped_ambiguous || 0}`,
      `失败 ${counts.verify_failed || 0}`,
      `异常 ${counts.error || 0}`,
      `复跑 ${state.retryProcessed || 0}`
    ].join(' ｜ ');
    if (panelEls.summary) panelEls.summary.textContent = text;
  }

  function setMode(text, type = 'idle') {
    if (panelEls.mode) {
      panelEls.mode.textContent = text;
      panelEls.mode.dataset.type = type;
    }
  }

  function refreshButtons() {
    if (!panelEls.start) return;
    panelEls.start.disabled = state.running;
    panelEls.pause.disabled = !state.running || state.waitingManual;
    panelEls.stop.disabled = !state.running && !state.waitingManual;
    if (panelEls.next) panelEls.next.disabled = !state.waitingManual;
    panelEls.clearInput.disabled = state.running || state.waitingManual;
    panelEls.pause.textContent = state.pauseRequested ? '继续' : '暂停';
  }

  function ensurePageReadyBeforeRun() {
    if (!/\/account\/index/.test(location.href)) {
      log('当前 URL 可能不是客户列表页，请确认在 CRM 客户列表运行。', 'warn');
    }
    if (!findSearchInput()) {
      throw new Error(`找不到“通过名称搜索”搜索框：${CRM_PATHS.searchInput}`);
    }
    if (!document.querySelector(CRM_PATHS.dataTable)) {
      throw new Error(`找不到客户数据表：${CRM_PATHS.dataTable}`);
    }
    if (findEditForm()) {
      throw new Error('当前存在未关闭的编辑表单，请先手动取消或保存后再运行。');
    }
    const tableState = readTableState();
    if (!tableState.headers.length) {
      throw new Error(`当前页面未读到客户表格表头：${CRM_PATHS.headerTh}`);
    }
    log('路径自检通过：搜索框、客户表格、精确列路径可用。', 'ok');
    if (hasActiveFilterHint()) {
      log('检测到页面存在“清空筛选”。脚本不会自动清空筛选；若后续搜索 0 条，可能与筛选有关。', 'warn');
    }
  }

  async function waitIfPaused() {
    while (state.running && state.pauseRequested && !state.stopRequested) {
      setMode('暂停中', 'paused');
      await sleep(300);
    }
    if (state.running && !state.stopRequested) setMode('运行中', 'running');
  }

  async function run() {
    if (state.running) return;

    resetRunLogs();

    let parsed;
    try {
      parsed = parseInput();
    } catch (err) {
      log(err.message || String(err), 'error');
      return;
    }

    if (parsed.errors.length) {
      parsed.errors.slice(0, 20).forEach(msg => log(msg, 'error'));
      if (parsed.errors.length > 20) log(`还有 ${parsed.errors.length - 20} 条输入错误未显示。`, 'error');
      log('输入存在错误，已停止。请修正后重新开始。', 'error');
      return;
    }

    const targets = parsed.targets;
    if (!targets.length) {
      log('没有识别到可处理数据。请从 WPS 复制“客户名称”到“rating”的列区域，包含表头或内容均可。', 'error');
      return;
    }

    try {
      ensurePageReadyBeforeRun();
    } catch (err) {
      log(err.message || String(err), 'error');
      return;
    }

    state.running = true;
    state.pauseRequested = false;
    state.stopRequested = false;
    state.total = targets.length;
    state.current = 0;
    setMode('运行中', 'running');
    refreshButtons();
    refreshSummary();

    log(`开始处理：${targets.length} 条。输入识别方式：${parsed.indexSource || '自动'}`, 'info');
    log('运行规则：只写入“行业”和“客户优先级”；每次保存后都会回到列表回读验证。', 'info');

    for (const target of targets) {
      await waitIfPaused();
      if (state.stopRequested) {
        log('已收到停止请求，当前批次中止。', 'warn');
        break;
      }

      state.current += 1;
      state.activeCompanyName = target.name;
      refreshSummary();

      try {
        log(`处理 ${state.current}/${state.total}：${target.name} -> 行业=${target.industry} 优先级=${target.priority}`, 'info');
        const result = await processTarget(target);
        state.results.push(result);
        const level = ['updated_verified', 'already_match_verified'].includes(result.status)
          ? (result.note ? 'warn' : 'ok')
          : (result.status === 'verify_failed' || result.status === 'error' ? 'error' : 'warn');
        log(`${target.name}：${summarizeResult(result)}`, level);
      } catch (err) {
        await cancelFormIfPresent().catch(() => {});
        const result = {
          index: target.lineNumber,
          name: target.name,
          industry: target.industry,
          priority: target.priority,
          status: 'error',
          error: err && err.stack ? err.stack : String(err)
        };
        state.results.push(result);
        log(`${target.name}：异常 ${err.message || err}`, 'error');
      }

      state.activeCompanyName = '';
      refreshSummary();
      await sleep(120);
    }

    state.activeCompanyName = '';

    await retryWarnTargets(targets);

    try {
      await clearSearch();
      log('已清空搜索框。', 'info');
    } catch (err) {
      log(`清空搜索框失败：${err.message || err}`, 'warn');
    }

    state.running = false;
    state.pauseRequested = false;
    state.stopRequested = false;
    state.waitingManual = false;
    state.manualContinueResolve = null;
    setMode('空闲', 'idle');
    refreshButtons();
    refreshSummary();
    log('处理结束。', 'info');
  }

  function clamp(value, min, max) {
    return Math.max(min, Math.min(max, value));
  }

  function applySavedPanelPosition(panel) {
    const defaultWidth = 430;
    const defaultLeft = Math.max(12, window.innerWidth - defaultWidth - 18);
    const defaultTop = 76;
    let pos = null;
    try {
      pos = JSON.parse(localStorage.getItem(STORAGE_KEYS.panelPos) || 'null');
    } catch (_) {}
    const left = clamp(Number(pos?.left ?? defaultLeft), 6, Math.max(6, window.innerWidth - 120));
    const top = clamp(Number(pos?.top ?? defaultTop), 6, Math.max(6, window.innerHeight - 36));
    panel.style.left = `${left}px`;
    panel.style.top = `${top}px`;
    panel.style.right = 'auto';
    panel.style.bottom = 'auto';
  }

  function savePanelPosition(panel) {
    const rect = panel.getBoundingClientRect();
    localStorage.setItem(STORAGE_KEYS.panelPos, JSON.stringify({ left: Math.round(rect.left), top: Math.round(rect.top) }));
  }

  function initDrag(panel, handle) {
    let dragging = false;
    let startX = 0;
    let startY = 0;
    let startLeft = 0;
    let startTop = 0;

    handle.addEventListener('pointerdown', event => {
      if (event.button !== 0) return;
      if (event.target.closest('button, input, textarea, select, a')) return;
      const rect = panel.getBoundingClientRect();
      dragging = true;
      startX = event.clientX;
      startY = event.clientY;
      startLeft = rect.left;
      startTop = rect.top;
      panel.style.left = `${rect.left}px`;
      panel.style.top = `${rect.top}px`;
      panel.style.right = 'auto';
      panel.style.bottom = 'auto';
      handle.setPointerCapture?.(event.pointerId);
      event.preventDefault();
    });

    handle.addEventListener('pointermove', event => {
      if (!dragging) return;
      const dx = event.clientX - startX;
      const dy = event.clientY - startY;
      const rect = panel.getBoundingClientRect();
      const maxLeft = window.innerWidth - Math.min(120, rect.width);
      const maxTop = window.innerHeight - 34;
      const nextLeft = clamp(startLeft + dx, 6, maxLeft);
      const nextTop = clamp(startTop + dy, 6, maxTop);
      panel.style.left = `${nextLeft}px`;
      panel.style.top = `${nextTop}px`;
      event.preventDefault();
    });

    const endDrag = event => {
      if (!dragging) return;
      dragging = false;
      savePanelPosition(panel);
      try { handle.releasePointerCapture?.(event.pointerId); } catch (_) {}
    };
    handle.addEventListener('pointerup', endDrag);
    handle.addEventListener('pointercancel', endDrag);
  }

  function minimizePanel(minimized) {
    panelEls.panel.classList.toggle('crm-minimized', minimized);
    panelEls.minimize.textContent = minimized ? '□' : '—';
    panelEls.minimize.title = minimized ? '展开' : '最小化';
    localStorage.setItem(STORAGE_KEYS.minimized, minimized ? '1' : '0');

  }

    if (state.running) {
      log('脚本正在运行，请先暂停或停止后再关闭面板。', 'warn');

    panelEls.launcher.style.top = `${Math.round(rect.top)}px`;

    panelEls.launcher.style.display = 'flex';
  }

    panelEls.panel.style.display = 'block';
    panelEls.launcher.style.display = 'none';
    applySavedPanelPosition(panelEls.panel);
  }
