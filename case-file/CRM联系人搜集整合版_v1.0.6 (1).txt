(function () {
  'use strict';

  const VERSION = '1.0.5';
  const PANEL_ID = 'crm-contact-collect-panel-v105';
  const STYLE_ID = 'crm-contact-collect-style-v105';

  const COLLECT_MAP = {
    done: '已做',
    linkedin: '只有领英无邮箱',
    none: '皆无',
    sn: 'SN',
    not_done: '未做'
  };

  const state = {
    companies: [],
    index: 0,
    running: false,
    stopped: false,
    busy: false,
    waitingNext: false,
    waitingManualSearch: false,
    lastFieldText: '',
    highlightedItem: null,
    panel: null,
    refs: {},
    updateTimer: null,
    currentCompany: '',
    currentPage: 'unknown',
    minimized: false,
    lastCopiedText: '',
    readonlyHintAt: 0,
    manualHintAt: 0,
  };

  function sleep(ms) {
    return new Promise(resolve => setTimeout(resolve, ms));
  }

  function log(msg) {
    const now = new Date();
    const ts = `${String(now.getHours()).padStart(2, '0')}:${String(now.getMinutes()).padStart(2, '0')}:${String(now.getSeconds()).padStart(2, '0')}`;
    const line = `[${ts}] ${msg}`;
    if (state.refs.log) {
      const previous = state.refs.log.textContent || '';
      state.refs.log.textContent = previous ? `${previous}
${line}` : line;
      state.refs.log.scrollTop = state.refs.log.scrollHeight;
    }
  }

  function normalizeText(text) {
    return String(text || '')
      .normalize('NFKC')
      .replace(/\s+/g, ' ')
      .trim()
      .toLowerCase();
  }

  function normalizeCompanyName(text) {
    return normalizeText(text)
      .replace(/[®™©℠]/g, '')
      .replace(/[()（）【】\[\]{}<>《》'"`´’‘“”.,，。:：;；!！?？·•|｜\\/]+/g, ' ')
      .replace(/\s+/g, ' ')
      .trim();
  }

  function isVisible(el) {
    if (!el || !(el instanceof Element)) return false;
    if (isInsidePanel(el)) return false;
    const style = window.getComputedStyle(el);
    const rect = el.getBoundingClientRect();
    return style.display !== 'none' && style.visibility !== 'hidden' && style.opacity !== '0' && rect.width > 0 && rect.height > 0;
  }

  function getPanelRoot() {
    return document.getElementById(PANEL_ID);
  }

  function isInsidePanel(el) {
    const panel = getPanelRoot();
    return !!(panel && el instanceof Node && panel.contains(el));
  }

  function queryVisibleAll(selector, root = document) {
    return Array.from(root.querySelectorAll(selector)).filter(isVisible);
  }

  function setNativeValue(el, value) {
    const proto = Object.getPrototypeOf(el);
    const desc = Object.getOwnPropertyDescriptor(proto, 'value');
    if (desc && desc.set) desc.set.call(el, value);
    else el.value = value;
    el.dispatchEvent(new Event('input', { bubbles: true }));
    el.dispatchEvent(new Event('change', { bubbles: true }));
  }

  function pressEnter(el) {
    const init = { key: 'Enter', code: 'Enter', keyCode: 13, which: 13, bubbles: true, cancelable: true, composed: true };
    el.dispatchEvent(new KeyboardEvent('keydown', init));
    el.dispatchEvent(new KeyboardEvent('keypress', init));
    el.dispatchEvent(new KeyboardEvent('keyup', init));
  }

  function mouseClick(el) {
    if (!el) return false;
    const rect = el.getBoundingClientRect();
    const clientX = rect.left + Math.max(6, Math.min(rect.width - 6, rect.width / 2));
    const clientY = rect.top + Math.max(6, Math.min(rect.height - 6, rect.height / 2));
    const realWin = el.ownerDocument?.defaultView || window;
    for (const type of ['mouseover', 'mousedown', 'mouseup', 'click']) {
      try {
        el.dispatchEvent(new realWin.MouseEvent(type, {
          bubbles: true,
          cancelable: true,
          clientX,
          clientY,
        }));
      } catch (_) {
        try {
          el.dispatchEvent(new MouseEvent(type, {
            bubbles: true,
            cancelable: true,
            clientX,
            clientY,
          }));
        } catch (_) {}
      }
    }
    try { el.click(); } catch (_) {}
    return true;
  }

  async function waitFor(fn, timeout = 10000, interval = 200, desc = '目标') {
    const start = Date.now();
    while (Date.now() - start < timeout) {
      try {
        const result = fn();
        if (result) return result;
      } catch (_) {}
      await sleep(interval);
    }
    throw new Error(`等待超时：${desc}`);
  }

  function findSearchInput() {
    return queryVisibleAll('input.n-input__input-el[placeholder="通过名称搜索"], input[placeholder="通过名称搜索"]')[0] || null;
  }

  function findNoDataText() {
    const preferred = queryVisibleAll('.n-data-table-empty, .empty, .no-data').find(el => {
      return normalizeText(el.textContent).includes('暂无数据');
    });
    if (preferred) return preferred;

    const textNodes = Array.from(document.querySelectorAll('span, div, td, p'));
    return textNodes.find(el => {
      if (!isVisible(el)) return false;
      const text = normalizeText(el.textContent || '');
      if (text !== '暂无数据') return false;
      const rect = el.getBoundingClientRect();
      return rect.width > 20 && rect.height > 12;
    }) || null;
  }

  function isMainPage() {
    return !!findSearchInput();
  }

  function findBackButton() {
    const buttons = queryVisibleAll('button.n-button, button');
    for (const btn of buttons) {
      const path = btn.querySelector('svg path');
      const d = path?.getAttribute('d') || '';
      if (d.includes('M328 112L184 256l144 144')) return btn;
    }
    return null;
  }

  function findDetailAttachmentField() {
    const rows = queryVisibleAll('.field-line');
    for (const row of rows) {
      const text = normalizeText(row.textContent || '');
      if (!text.includes('附件')) continue;
      const button = row.querySelector('button .n-button__content, button');
      return { row, button: button ? (button.closest('button') || button) : null };
    }
    return null;
  }

  function findDetailDescriptionItemByLabel(labelText) {
    const items = queryVisibleAll('.crm-description-item');
    for (const item of items) {
      const labelEl = item.querySelector('.crm-description-item-label .one-line-text, .crm-description-item-label');
      if (!labelEl) continue;
      if (normalizeText(labelEl.textContent || '') === normalizeText(labelText)) return item;
    }
    return null;
  }

  function getDetailDisplayValueByLabel(labelText) {
    const item = findDetailDescriptionItemByLabel(labelText);
    if (!item) return null;
    const valueEl = item.querySelector('.crm-description-item-value .one-line-text, .crm-description-item-value');
    if (!valueEl) return '';
    return (valueEl.textContent || '').trim();
  }

  function getCollectFieldDisplayText() {
    const raw = getDetailDisplayValueByLabel('是否联系人收集');
    if (raw == null) return 'null';
    if (!String(raw).trim() || String(raw).trim() === '-') return '未做';
    return String(raw).trim();
  }

  function findEditButton() {
    const byAttr = queryVisibleAll('button[label="编辑"]')[0];
    if (byAttr) return byAttr;
    const buttons = queryVisibleAll('button.n-button, button');
    return buttons.find(btn => normalizeText(btn.querySelector('.n-button__content')?.textContent || btn.textContent || '') === '编辑') || null;
  }

  function findUpdateButton() {
    const buttons = queryVisibleAll('button.n-button, button');
    return buttons.find(btn => {
      const text = normalizeText(btn.querySelector('.n-button__content')?.textContent || btn.textContent || '');
      return text === '更新';
    }) || null;
  }

  function findFormItemByLabel(labelText) {
    const labels = queryVisibleAll('.n-form-item .n-form-item-label__text');
    const match = labels.find(el => normalizeText(el.textContent) === normalizeText(labelText));
    return match ? match.closest('.n-form-item') : null;
  }

  function getInputValueByLabel(labelText) {
    const item = findFormItemByLabel(labelText);
    if (!item) return null;
    const input = item.querySelector('input.n-input__input-el');
    if (input) return input.value || '';
    const textarea = item.querySelector('textarea.n-input__textarea-el');
    if (textarea) return textarea.value || textarea.getAttribute('value') || '';
    return null;
  }

  function findContactCollectInput() {
    const item = findFormItemByLabel('是否联系人收集');
    return item?.querySelector('input.n-input__input-el[type="text"], input.n-input__input-el') || null;
  }

  function getCustomerNameValue() {
    return getInputValueByLabel('客户名称');
  }

  function isEditPage() {
    return !!findFormItemByLabel('客户名称');
  }

  function isEditPageReady() {
    const name = getCustomerNameValue();
    return typeof name === 'string' && name.trim() !== '';
  }

  function isDetailPage() {
    return !!(findEditButton() && findBackButton());
  }

  function detectPageType() {
    if (isEditPage()) return 'edit';
    if (isDetailPage()) return 'detail';
    if (isMainPage()) return 'main';
    return 'other';
  }

  function clearHighlight() {
    if (!state.highlightedItem) return;
    state.highlightedItem.classList.remove('crm-contact-highlight-target', 'crm-contact-highlight-blink');
    state.highlightedItem = null;
  }

  function markHighlight(targetEl) {
    clearHighlight();
    targetEl.classList.add('crm-contact-highlight-target', 'crm-contact-highlight-blink');
    state.highlightedItem = targetEl;
    setTimeout(() => {
      if (state.highlightedItem === targetEl) targetEl.classList.remove('crm-contact-highlight-blink');
    }, 1700);
  }

  function findScrollableParent(el) {
    let parent = el?.parentElement || null;
    while (parent && parent !== document.body && parent !== document.documentElement) {
      const style = window.getComputedStyle(parent);
      const canScroll = /(auto|scroll|overlay)/.test(style.overflowY) && parent.scrollHeight > parent.clientHeight + 8;
      if (canScroll) return parent;
      parent = parent.parentElement;
    }
    return null;
  }

  async function scrollIntoMiddle(targetEl) {
    if (!targetEl) return;
    const scroller = findScrollableParent(targetEl);
    if (scroller) {
      const sr = scroller.getBoundingClientRect();
      const tr = targetEl.getBoundingClientRect();
      const nextTop = scroller.scrollTop + (tr.top - sr.top) - (scroller.clientHeight / 2) + (tr.height / 2);
      try {
        scroller.scrollTo({ top: Math.max(0, nextTop), behavior: 'smooth' });
      } catch (_) {
        scroller.scrollTop = Math.max(0, nextTop);
      }
    } else {
      try {
        targetEl.scrollIntoView({ behavior: 'smooth', block: 'center', inline: 'nearest' });
      } catch (_) {
        targetEl.scrollIntoView(true);
      }
    }
    await sleep(500);
  }

  function scrollToPageBottom() {
    const candidates = Array.from(document.querySelectorAll('.n-scrollbar-container, .n-scrollbar-content, .n-scrollbar, [class*="scroll"]'))
      .filter(el => el instanceof HTMLElement && !isInsidePanel(el));
    let best = null;
    let maxScrollable = 0;
    for (const el of candidates) {
      const scrollable = el.scrollHeight - el.clientHeight;
      if (scrollable > 20 && scrollable > maxScrollable) {
        best = el;
        maxScrollable = scrollable;
      }
    }
    if (best) {
      best.scrollTop = best.scrollHeight;
      return { type: 'container' };
    }
    window.scrollTo(0, Math.max(document.body.scrollHeight, document.documentElement.scrollHeight));
    return { type: 'window' };
  }

  function getMainResultRows() {
    const rows = queryVisibleAll('table tbody tr, .n-data-table-tbody tr');
    return rows.filter(row => row.querySelector('td[data-col-key="name"]'));
  }

  function getClickableCompanyEntries() {
    const rows = getMainResultRows();
    const entries = [];
    for (const row of rows) {
      const td = row.querySelector('td[data-col-key="name"]');
      if (!td || !isVisible(td)) continue;
      const btn = td.querySelector('button');
      const textEl = td.querySelector('.one-line-text') || td.querySelector('.n-button__content') || td;
      const text = (textEl?.textContent || '').trim();
      if (!text) continue;
      if (!btn || !isVisible(btn)) continue;
      entries.push({ row, td, btn, text, normalized: normalizeCompanyName(text) });
    }
    return entries;
  }

  async function ensureMainPage() {
    const page = detectPageType();
    if (page === 'main') return true;
    if (page === 'detail') {
      const backBtn = findBackButton();
      if (!backBtn) throw new Error('当前在详情页，但未找到返回按钮。');
      log('当前在详情页，先自动返回主列表。');
      mouseClick(backBtn);
      await waitFor(() => findSearchInput(), 10000, 200, '主列表搜索框');
      return true;
    }
    throw new Error('当前不在主列表/详情/编辑可控页面，需人工先返回主页面。');
  }

  function setPausedForManual(reason) {
    state.running = false;
    state.waitingNext = false;
    state.waitingManualSearch = true;
    updateStatus(reason);
    log(reason);
  }

  async function searchCompany(companyName) {
    const input = await waitFor(() => findSearchInput(), 10000, 200, '搜索框');
    input.focus();
    input.select?.();
    setNativeValue(input, '');
    await sleep(100);
    setNativeValue(input, companyName);
    await sleep(120);
    pressEnter(input);
    log(`Step.1 搜索公司：${companyName}`);
    await sleep(1200);

    await waitFor(() => {
      if (findNoDataText()) return true;
      const rows = getMainResultRows();
      const entries = getClickableCompanyEntries();
      return rows.length > 0 || entries.length > 0 || !!findNoDataText();
    }, 8000, 250, '搜索结果变化').catch(() => null);

    if (findNoDataText()) {
      return { status: 'NO_DATA', entries: [] };
    }

    const entries = getClickableCompanyEntries();
    if (!entries.length) {
      return { status: 'NO_CLICKABLE', entries };
    }

    const target = normalizeCompanyName(companyName);
    const matched = entries.filter(x => x.normalized === target || x.normalized.includes(target) || target.includes(x.normalized));

    if (matched.length > 1) {
      return { status: 'MULTI_MATCH', entries: matched };
    }
    if (matched.length === 1) {
      return { status: 'ONE_MATCH', entries: matched };
    }
    if (entries.length === 1) {
      return { status: 'ONE_ROW', entries };
    }
    return { status: 'MULTI_ROW', entries };
  }

  async function openCompanyFromSearch(companyName) {
    const result = await searchCompany(companyName);

    if (result.status === 'NO_DATA') {
      setPausedForManual(`搜索结果为“暂无数据”，已暂停：${companyName}`);
      return false;
    }
    if (result.status === 'NO_CLICKABLE') {
      setPausedForManual(`搜索后未找到可点击公司名，已暂停：${companyName}`);
      return false;
    }
    if (result.status === 'MULTI_MATCH' || result.status === 'MULTI_ROW') {
      setPausedForManual(`搜索结果存在多条可点击记录，已暂停等待人工处理：${companyName}`);
      return false;
    }

    const entry = result.entries[0];
    mouseClick(entry.btn);
    log(`Step.2 已点击公司：${companyName}`);
    await waitFor(() => isDetailPage(), 10000, 200, '详情页');
    return true;
  }

  async function goToEditPage(companyName) {
    const editBtn = await waitFor(() => findEditButton(), 10000, 200, '编辑按钮');
    mouseClick(editBtn);
    log(`Step.3 已点击编辑：${companyName}`);
    await waitFor(() => isEditPage(), 10000, 200, '编辑页');
    await waitFor(() => isEditPageReady(), 12000, 250, '编辑页客户名称加载完成');
    log(`Step.4 编辑页已加载完成。客户名称：${getCustomerNameValue() || '-'}`);
    await autoCopyWebsiteOnEditEnter();
    return true;
  }

  async function copyTextToClipboard(text, label, options = {}) {
    const content = String(text ?? '');
    state.lastCopiedText = content;
    const source = options.source || 'manual';

    if (typeof GM_setClipboard === 'function') {
      try {
        GM_setClipboard(content, 'text');
        log(`已复制${label}：${content || '(空)'} ｜ 方式：GM_setClipboard ｜ 来源：${source}`);
        return { ok: true, method: 'GM_setClipboard' };
      } catch (err) {
        log(`复制${label}时 GM_setClipboard 失败：${err?.message || err}`);
      }
    } else {
      log(`复制${label}时无法使用 GM_setClipboard：当前脚本环境未授予该能力。`);
    }

    if (window.isSecureContext && navigator.clipboard?.writeText) {
      try {
        await navigator.clipboard.writeText(content);
        log(`已复制${label}：${content || '(空)'} ｜ 方式：navigator.clipboard ｜ 来源：${source}`);
        return { ok: true, method: 'navigator.clipboard' };
      } catch (err) {
        log(`复制${label}时 navigator.clipboard 失败：${err?.message || err}`);
      }
    } else {
      log(`复制${label}时无法使用 navigator.clipboard：页面非安全上下文，或浏览器不支持。`);
    }

    try {
      const ta = document.createElement('textarea');
      ta.value = content;
      ta.setAttribute('readonly', 'readonly');
      ta.style.position = 'fixed';
      ta.style.left = '12px';
      ta.style.top = '12px';
      ta.style.width = '1px';
      ta.style.height = '1px';
      ta.style.opacity = '0.01';
      ta.style.zIndex = '2147483647';
      document.body.appendChild(ta);
      ta.focus();
      ta.select();
      ta.setSelectionRange(0, ta.value.length);
      const ok = document.execCommand('copy');
      ta.remove();
      if (!ok) throw new Error('document.execCommand 返回 false');
      log(`已复制${label}：${content || '(空)'} ｜ 方式：execCommand ｜ 来源：${source}`);
      return { ok: true, method: 'execCommand' };
    } catch (err) {
      log(`复制${label}失败：${err?.message || err}。最近一次复制内容已缓存到面板内存，可继续排查油猴权限或浏览器策略。`);
      return { ok: false, method: 'failed', error: err?.message || String(err) };
    }
  }

  async function handleCopy(labelText) {

      log(`当前不在编辑页，不能复制“${labelText}”。`);
      return;
    }
    const value = getInputValueByLabel(labelText);
    if (value == null) {
      log(`未找到字段：${labelText}`);
      return;
    }
    await copyTextToClipboard(value, labelText, { source: 'manual-click' });
  }

  async function fillCollectAndUpdate(mode) {

  async function reopenCurrentCompanyDetailForAttachmentRetry() {
    const companyName = state.currentCompany || state.companies[state.index];
    if (!companyName) throw new Error('缺少当前公司名，无法重进详情页。');

    const backBtn = await waitFor(() => findBackButton(), 5000, 200, '详情页返回按钮');

    log('附件字段首次缺失：已返回主列表，准备重新进入当前公司详情页。');

    const reopened = await openCompanyFromSearch(companyName);
    if (!reopened) {
      throw new Error('重进详情页失败：当前公司搜索结果不稳定，请人工处理。');
    }
    log('已重新进入当前公司详情页，开始再次查找附件字段。');
  }

  async function focusAttachmentField() {
    let target = await waitFor(() => findDetailAttachmentField(), 5500, 200, '附件字段').catch(() => null);

    if (!target) {
      log('附件字段首次未显示：尝试自动返回并重新进入详情页。');
      await reopenCurrentCompanyDetailForAttachmentRetry();
      target = await waitFor(() => findDetailAttachmentField(), 7000, 200, '附件字段（二次查找）').catch(() => null);
    }

    if (!target) {
      throw new Error('附件字段两次查找仍失败，请人工检查 CRM 页面加载情况。');
    }

    await scrollIntoMiddle(target.row);
    markHighlight(target.row);
    const text = (target.button?.textContent || target.row.textContent || '').trim();
    state.lastFieldText = text || '(空)';
    log(`Step.7 已聚焦到附件字段。`);
    log(`字段：${state.lastFieldText}`);
    updateStatus();
  }

  async function processCurrentCompany() {
    const companyName = state.companies[state.index];
    state.currentCompany = companyName;
    state.lastFieldText = '';
    updateStatus();

    await ensureMainPage();
    const opened = await openCompanyFromSearch(companyName);
    if (!opened) return;

    state.lastFieldText = getCollectFieldDisplayText();
    updateStatus();
    log(`Step.3 当前“是否联系人收集” = ${state.lastFieldText}`);

    await goToEditPage(companyName);
    state.waitingNext = false;
    log('已进入编辑页。请先处理复制/填写；完成当前公司后，点击“下一个”继续。');
  }

  async function processNext() {
    if (state.busy || !state.running || state.stopped) return;
    if (state.index >= state.companies.length) {
      state.running = false;
      state.waitingNext = false;
      state.waitingManualSearch = false;
      state.lastFieldText = '全部完成';
      updateStatus();
      log('全部公司处理完成。');
      return;
    }

    state.busy = true;
    try {
      log('------------------------------');
      log(`开始处理：${state.companies[state.index]}`);
      await processCurrentCompany();
    } catch (err) {
      log(`处理失败：${err.message || err}`);
      state.running = false;
      state.waitingNext = false;
      updateStatus('已暂停，请人工检查。');
    } finally {
      state.busy = false;
    }
  }

  function parseCompanies(raw) {
    return String(raw || '')
      .split(/\r?\n/)
      .map(s => s.trim())
      .filter(Boolean);
  }

  async function handleStart() {
    const companies = parseCompanies(state.refs.input.value || '');
    if (!companies.length) {
      alert('请先粘贴公司名单，每行一个。');
      return;
    }
    state.companies = companies;
    state.index = 0;
    state.running = true;
    state.stopped = false;
    state.waitingNext = false;
    state.waitingManualSearch = false;
    state.lastFieldText = '';
    clearHighlight();
    updateStatus();
    log(`已载入 ${companies.length} 家公司，开始执行。`);
    await processNext();
  }

  function handleStop() {
    state.stopped = true;
    state.running = false;
    state.waitingNext = false;
    state.waitingManualSearch = false;
    clearHighlight();
    updateStatus();
    log('已停止。');
  }

  async function handleNext() {
    if (state.busy) return;
    const page = detectPageType();

    try {
      if (page === 'edit') {
        log('当前仍在编辑页，请先点击“已做/领英/皆无/未做/填写SN”完成更新后再下一个。');
        return;
      }

      if (page === 'detail') {
        const backBtn = await waitFor(() => findBackButton(), 8000, 200, '详情页返回按钮');
        clearHighlight();
        mouseClick(backBtn);
        log('已从详情页返回主列表。');
        await waitFor(() => isMainPage(), 10000, 200, '主列表');
      } else if (page !== 'main') {
        log('当前不在可自动返回的详情页，需人工先返回主页面。');
        return;
      }

      state.index += 1;
      state.waitingNext = false;
      state.waitingManualSearch = false;
      state.lastFieldText = '';
      updateStatus();

      state.running = true;
      await sleep(350);
      await processNext();
    } catch (err) {
      log(`切换下一家公司失败：${err.message || err}`);
    }
  }

  async function handleContinueCurrent() {
    if (state.busy) return;
    state.waitingManualSearch = false;
    state.running = true;
    updateStatus();
    log('继续当前公司。');
    await processNext();
  }

  async function handleCollectAction(mode) {
    if (state.busy) return;
    state.busy = true;
    try {
      await fillCollectAndUpdate(mode);
      state.running = false;
      state.waitingNext = true;
      updateStatus();
      log('当前公司已完成。请点击“下一个”继续。');
    } catch (err) {
      log(`执行失败：${err.message || err}`);
      updateStatus('执行失败');
    } finally {
      state.busy = false;
    }
  }

  function updateInputReadonly() {
    if (!state.refs.input) return;
    const lock = detectPageType() !== 'main';
    state.refs.input.readOnly = lock;
    state.refs.input.classList.toggle('crm-contact-readonly', lock);
  }

  function updateStatus(extra = '') {
    if (!state.panel || !document.body.contains(state.panel)) return;
    updateInputReadonly();
    state.currentPage = detectPageType();
    const total = state.companies.length;
    const current = total === 0 ? 0 : Math.min(state.index + 1, total);
    const name = state.companies[state.index] || '-';
    let stage = '空闲';
    if (state.busy) stage = '运行中';
    else if (state.waitingManualSearch) stage = '等待人工处理';
    else if (state.waitingNext) stage = '等待下一个';
    else if (state.running) stage = '运行中';

    state.refs.status.textContent = `进度：${current}/${total} ｜ 状态：${stage} ｜ 当前：${name}`;
    state.refs.value.textContent = `字段：${state.lastFieldText || '-'} ｜ 页面：${state.currentPage}${extra ? ` ｜ ${extra}` : ''}`;
    state.refs.start.disabled = state.running || state.busy;
    state.refs.stop.disabled = (!state.running && !state.waitingManualSearch && !state.waitingNext) || state.busy;
    state.refs.next.disabled = state.busy;
    state.refs.continueCurrent.disabled = !(state.waitingManualSearch || detectPageType() === 'main') || state.busy;
    state.refs.end.disabled = state.busy;

    const copyDisabled = !isEditPageReady() || state.busy;
    state.refs.copyName.disabled = copyDisabled;
    state.refs.copyWebsite.disabled = copyDisabled;
    state.refs.copyIntro.disabled = copyDisabled;

    const actionDisabled = !isEditPageReady() || state.busy;
    state.refs.actionButtons.forEach(btn => { btn.disabled = actionDisabled; });

    if (state.refs.minimize) {
      state.refs.minimize.textContent = state.minimized ? '＋' : '－';
      state.refs.minimize.title = state.minimized ? '展开' : '最小化';
    }
    if (state.panel) {
      state.panel.classList.toggle('crm-minimized', !!state.minimized);
    }
    if (state.refs.body) {
      state.refs.body.style.display = state.minimized ? 'none' : '';
    }
  }

  function makeDraggable(panel, handle) {
    let dragging = false;
    let startX = 0, startY = 0, startRight = 20, startBottom = 20;

    handle.addEventListener('mousedown', e => {
      if (e.target.closest('button, textarea, input, label')) return;
      dragging = true;
      startX = e.clientX;
      startY = e.clientY;
      const rect = panel.getBoundingClientRect();
      startRight = window.innerWidth - rect.right;
      startBottom = window.innerHeight - rect.bottom;
      document.body.style.userSelect = 'none';
    });

    window.addEventListener('mousemove', e => {
      if (!dragging) return;
      const dx = e.clientX - startX;
      const dy = e.clientY - startY;
      panel.style.right = `${Math.max(0, startRight - dx)}px`;
      panel.style.bottom = `${Math.max(0, startBottom - dy)}px`;
      panel.style.left = 'auto';
      panel.style.top = 'auto';
    });

    window.addEventListener('mouseup', () => {
      if (!dragging) return;
      dragging = false;
      document.body.style.userSelect = '';
    });
  }
